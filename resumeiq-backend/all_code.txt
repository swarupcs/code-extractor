import "dotenv/config";

import express, { type Request, type Response } from "express";
import cors from "cors";
import cookieParser from "cookie-parser";
import helmet from "helmet";
import compression from "compression";
import { fileURLToPath } from "url";
import path from "path";

import connectDB from "./config/db.js";
import { initializePassport } from "./config/passport.js";
import { globalErrorHandler } from "./middlewares/error.middleware.js";
import apiRouter from "./routes/api.routes.js";

const app  = express();
const PORT = Number(process.env["PORT"] ?? 8080);

// ── Database ──────────────────────────────────────────────────────────────────
await connectDB();

// ── Passport ──────────────────────────────────────────────────────────────────
initializePassport();

// ── Security headers ──────────────────────────────────────────────────────────
app.use(
  helmet({
    crossOriginEmbedderPolicy: false, // allow Puppeteer iframe embedding
    contentSecurityPolicy: false,     // CSP managed per-route if needed
  })
);

// ── GZIP compression ──────────────────────────────────────────────────────────
app.use(compression());

// ── CORS ──────────────────────────────────────────────────────────────────────
const allowedOrigins = (process.env["CLIENT_URL"] ?? "")
  .split(",")
  .map((o) => o.trim())
  .filter(Boolean);

app.use(
  cors({
    origin: (origin, callback) => {
      if (!origin) return callback(null, true);
      if (allowedOrigins.includes(origin)) return callback(null, true);
      return callback(new Error(`Origin "${origin}" not allowed by CORS`));
    },
    credentials:    true,
    methods:        ["GET", "POST", "PUT", "PATCH", "DELETE", "OPTIONS"],
    allowedHeaders: ["Content-Type", "Authorization"],
  })
);

app.options("/{*wildcard}", cors());

// ── Body parsing ──────────────────────────────────────────────────────────────
// 2 MB limit to accommodate LinkedIn/resume text paste payloads
app.use(express.json({ limit: "2mb" }));
app.use(express.urlencoded({ extended: true, limit: "2mb" }));
app.use(cookieParser());

// ── Static files (robots.txt etc.) ───────────────────────────────────────────
const __dirname = path.dirname(fileURLToPath(import.meta.url));
app.use(express.static(path.join(__dirname, "../public")));

// ── Routes ────────────────────────────────────────────────────────────────────
app.get("/", (_req: Request, res: Response) => {
  res.json({ message: "Welcome to AI Resume Builder API", status: 200 });
});

app.use("/api", apiRouter);

// ── Global error handler ──────────────────────────────────────────────────────
app.use(globalErrorHandler);

// ── Unhandled rejection / exception guards ────────────────────────────────────
process.on("unhandledRejection", (reason: unknown) => {
  console.error("[UnhandledRejection]", reason);
});

process.on("uncaughtException", (err: Error) => {
  console.error("[UncaughtException]", err);
  setTimeout(() => process.exit(1), 500);
});

// ── Start ─────────────────────────────────────────────────────────────────────
app.listen(PORT, () => {
  console.log(`[Server] Running on port ${PORT} (${process.env["NODE_ENV"] ?? "development"})`);
});

export default app;

import passport from "passport";
import { Strategy as GoogleStrategy } from "passport-google-oauth20";
import User from "../models/user.model.js";

export const initializePassport = (): void => {
  const clientID = process.env["GOOGLE_CLIENT_ID"];
  const clientSecret = process.env["GOOGLE_CLIENT_SECRET"];
  const callbackURL = process.env["GOOGLE_CALLBACK_URL"];

  if (!clientID || !clientSecret) {
    console.warn(
      "[Passport] GOOGLE_CLIENT_ID or GOOGLE_CLIENT_SECRET not set — " +
        "Google OAuth will not be available."
    );
    return;
  }

  passport.use(
    new GoogleStrategy(
      { clientID, clientSecret, callbackURL: callbackURL ?? "" },
      async (_accessToken, _refreshToken, profile, done) => {
        try {
          const email = profile.emails?.[0]?.value?.toLowerCase();
          const firstName = profile.name?.givenName ?? profile.displayName ?? "User";
          const lastName = profile.name?.familyName ?? "";

          if (!email) {
            return done(new Error("No email returned from Google"), false);
          }

          let user = await User.findOne({ googleId: profile.id });

          if (!user) {
            user = await User.findOne({ emailId: email });
            if (user) {
              user.googleId = profile.id;
              user.isVerified = true;
              await user.save();
            } else {
              user = await User.create({
                firstName,
                lastName,
                emailId: email,
                googleId: profile.id,
                isVerified: true,
                password: null,
              });
            }
          }

          return done(null, user);
        } catch (error) {
          return done(error as Error, false);
        }
      }
    )
  );

  passport.serializeUser((user, done) => {
    // eslint-disable-next-line @typescript-eslint/no-explicit-any
    done(null, (user as any)._id as string);
  });

  passport.deserializeUser(async (id: string, done) => {
    try {
      const user = await User.findById(id).select("-password -refreshToken");
      done(null, user);
    } catch (err) {
      done(err as Error, null);
    }
  });

  console.log("[Passport] Google OAuth strategy registered.");
};

export default passport;

import bcrypt from "bcryptjs";
import jwt from "jsonwebtoken";
import type { Request, Response } from "express";
import User from "../models/user.model.js";
import { ApiError } from "../utils/api-error.js";
import { ApiResponse } from "../utils/api-response.js";
import { asyncHandler } from "../utils/async-handler.js";
import { clearAuthCookie, clearRefreshCookie } from "../utils/cookies.js";
import { validateSignUpData } from "../utils/validation.js";
import { sendVerificationEmail, sendPasswordResetEmail } from "../utils/email.js";
import { issueTokens, generateSecureToken, hashToken } from "../services/auth.service.js";
import type { JwtPayload } from "../types/index.js";
import type { IUser } from "../models/user.model.js";

// ── Sign Up ───────────────────────────────────────────────────────────────────

export const signupUser = asyncHandler(async (req: Request, res: Response) => {
  const { firstName, lastName, emailId, password } = req.body as {
    firstName?: string; lastName?: string; emailId?: string; password?: string;
  };

  validateSignUpData(req);

  const existing = await User.findOne({ emailId });
  if (existing) throw new ApiError(409, "User already exists with this email");

  const hashedPassword = await bcrypt.hash(password!, 10);
  const { raw: verifyToken, hash: verifyHash } = generateSecureToken();

  const user = await User.create({
    firstName, lastName, emailId,
    password: hashedPassword,
    isVerified: false,
    emailVerificationToken: verifyHash,
    emailVerificationExpires: new Date(Date.now() + 24 * 60 * 60 * 1000),
  });

  sendVerificationEmail({ to: emailId!, firstName: firstName!, token: verifyToken }).catch(
    (err: unknown) => console.error("Verification email failed:", err)
  );

  await issueTokens(res, user);
  user.password = null;
  user.refreshToken = null;
  user.emailVerificationToken = null;

  return new ApiResponse(201, { user, requiresVerification: true },
    "Account created. Please check your email to verify your account.").send(res);
});

// ── Login ─────────────────────────────────────────────────────────────────────

export const loginUser = asyncHandler(async (req: Request, res: Response) => {
  const { emailId, password } = req.body as { emailId?: string; password?: string };
  if (!emailId || !password) throw new ApiError(400, "Email and password are required");

  const user = await User.findOne({ emailId });
  if (!user) throw new ApiError(401, "Invalid email or password");
  if (!user.password) throw new ApiError(400, "This account uses Google sign-in. Please continue with Google.");

  const isValid = user.comparePassword(password);
  if (!isValid) throw new ApiError(401, "Invalid email or password");

  await issueTokens(res, user);
  user.password = null;
  user.refreshToken = null;

  return new ApiResponse(200, { user }, "Login successful").send(res);
});

// ── Logout ────────────────────────────────────────────────────────────────────

export const logoutUser = asyncHandler(async (req: Request, res: Response) => {
  if (req.userId) await User.findByIdAndUpdate(req.userId, { refreshToken: null });
  clearAuthCookie(res);
  clearRefreshCookie(res);
  return new ApiResponse(200, null, "Logged out successfully").send(res);
});

// ── Refresh Tokens ────────────────────────────────────────────────────────────

export const refreshTokens = asyncHandler(async (req: Request, res: Response) => {
  const incomingRefresh = req.cookies?.refreshToken as string | undefined;
  if (!incomingRefresh) throw new ApiError(401, "No refresh token provided");

  const refreshSecret = process.env["JWT_REFRESH_SECRET"];
  if (!refreshSecret) throw new ApiError(500, "JWT_REFRESH_SECRET is not configured");

  let decoded: JwtPayload;
  try {
    decoded = jwt.verify(incomingRefresh, refreshSecret) as JwtPayload;
  } catch {
    throw new ApiError(401, "Invalid or expired refresh token");
  }

  const user = await User.findById(decoded.userId);
  if (!user?.refreshToken) throw new ApiError(401, "Session expired — please log in again");

  const isValid = await bcrypt.compare(incomingRefresh, user.refreshToken);
  if (!isValid) throw new ApiError(401, "Refresh token reuse detected — please log in again");

  await issueTokens(res, user);

  // Re-fetch user with sensitive fields excluded so we never accidentally leak them
  const freshUser = await User.findById(user._id).select("-password -refreshToken -emailVerificationToken -resetPasswordToken");

  return new ApiResponse(200, { user: freshUser }, "Tokens refreshed successfully").send(res);
});

// ── Verify Email ──────────────────────────────────────────────────────────────

export const verifyEmail = asyncHandler(async (req: Request, res: Response) => {
  const { token } = req.params as { token: string };
  if (!token) throw new ApiError(400, "Verification token is required");

  const hash = hashToken(token);
  const user = await User.findOne({ emailVerificationToken: hash, emailVerificationExpires: { $gt: new Date() } });
  if (!user) throw new ApiError(400, "Verification link is invalid or has expired");

  user.isVerified = true;
  user.emailVerificationToken = null;
  user.emailVerificationExpires = null;
  await user.save();

  return new ApiResponse(200, null, "Email verified successfully").send(res);
});

// ── Resend Verification ───────────────────────────────────────────────────────

export const resendVerification = asyncHandler(async (req: Request, res: Response) => {
  const { emailId } = req.body as { emailId?: string };
  if (!emailId) throw new ApiError(400, "Email is required");

  const genericMsg = "If your account exists and is unverified, a new verification email has been sent.";
  const user = await User.findOne({ emailId });

  const { raw: verifyToken, hash: verifyHash } = generateSecureToken();
  user.emailVerificationToken = verifyHash;
  user.emailVerificationExpires = new Date(Date.now() + 24 * 60 * 60 * 1000);
  await user.save();

  sendVerificationEmail({ to: user.emailId, firstName: user.firstName, token: verifyToken }).catch(

// ── Forgot Password ───────────────────────────────────────────────────────────

export const forgotPassword = asyncHandler(async (req: Request, res: Response) => {
  const { emailId } = req.body as { emailId?: string };
  if (!emailId) throw new ApiError(400, "Email is required");

  const genericMsg = "If an account with that email exists, a password reset link has been sent.";
  const user = await User.findOne({ emailId });
  if (!user || user.googleId) return new ApiResponse(200, null, genericMsg).send(res);

  const { raw: resetToken, hash: resetHash } = generateSecureToken();
  user.resetPasswordToken = resetHash;
  user.resetPasswordExpires = new Date(Date.now() + 60 * 60 * 1000);
  await user.save();

  sendPasswordResetEmail({ to: user.emailId, firstName: user.firstName, token: resetToken }).catch(
    (err: unknown) => console.error("Password reset email failed:", err)
  );

  return new ApiResponse(200, null, genericMsg).send(res);
});

// ── Reset Password ────────────────────────────────────────────────────────────

export const resetPassword = asyncHandler(async (req: Request, res: Response) => {
  const { token } = req.params as { token: string };
  const { password } = req.body as { password?: string };

  if (!token) throw new ApiError(400, "Reset token is required");
  if (!password) throw new ApiError(400, "New password is required");
  if (password.length < 8) throw new ApiError(400, "Password must be at least 8 characters");
  if (!/[A-Z]/.test(password)) throw new ApiError(400, "Password must contain at least one uppercase letter");
  if (!/\d/.test(password)) throw new ApiError(400, "Password must contain at least one number");

  const hash = hashToken(token);
  const user = await User.findOne({ resetPasswordToken: hash, resetPasswordExpires: { $gt: new Date() } });
  if (!user) throw new ApiError(400, "Password reset link is invalid or has expired");

  user.password = await bcrypt.hash(password, 10);
  user.resetPasswordToken = null;
  user.resetPasswordExpires = null;
  user.refreshToken = null;
  await user.save();

  clearAuthCookie(res);
  clearRefreshCookie(res);
  return new ApiResponse(200, null, "Password reset successful. Please log in with your new password.").send(res);
});

// ── Google OAuth Callback ─────────────────────────────────────────────────────

export const googleCallback = asyncHandler(async (req: Request, res: Response) => {
  const user = req.user as IUser | undefined;
  if (!user) throw new ApiError(401, "Google authentication failed");

  await issueTokens(res, user);

  const clientUrl = process.env["CLIENT_URL"]?.split(",")[0].trim() ?? "";
  res.redirect(`${clientUrl}/oauth/callback?success=true`);
});

      const shouldRemoveBg = removeBackground === true || removeBackground === "true";
      try {
        const imageStream = fs.createReadStream(image.path);
        const response = await imagekit.upload({
          file:     imageStream as unknown as string,
          fileName: `resume-${resumeId}.png`,
          folder:   "user-resumes",
          transformation: {
            pre: `w-300,h-300,fo-face,z-0.75${shouldRemoveBg ? ",e-bgremove" : ""}`,
          },
        });
        allowedUpdate.personal_info = { ...allowedUpdate.personal_info, image: response.url };
      } finally {
        if (fs.existsSync(image.path)) fs.unlinkSync(image.path);
      }
    }

    // Fetch current state to snapshot BEFORE overwriting
    const existing = await Resume.findOne({ _id: resumeId, userId });
    if (!existing) throw new ApiError(404, "Resume not found or access denied");

    const updatedVersions = pushVersion(existing);

    const resume = await Resume.findOneAndUpdate(
      { _id: resumeId, userId },
      { ...allowedUpdate, versions: updatedVersions },
      { new: true }
    );

    new ApiResponse(200, { resume }, "Resume saved successfully").send(res);
  }
);

  const { firstName, lastName, currentPassword, newPassword } = req.body as {

    currentPassword?: string;

  };

  // Update name fields if provided

  // Password change — only for email/password accounts

    if (!user.password) {

import jwt from "jsonwebtoken";
import type { Request, Response, NextFunction } from "express";
import User from "../models/user.model.js";
import { ApiError } from "../utils/api-error.js";
import type { JwtPayload } from "../types/index.js";

/**
 * Verifies the httpOnly JWT cookie and attaches req.userId.
 * All failures call next(err) so the globalErrorHandler processes them.
 */
export const authMiddleware = async (
  req: Request,
  _res: Response,
  next: NextFunction
): Promise<void> => {
  try {
    const token = req.cookies?.token as string | undefined;
    if (!token) throw new ApiError(401, "Please login first");

    const jwtSecret = process.env["JWT_SECRET"];
    if (!jwtSecret) throw new ApiError(500, "JWT_SECRET is not configured");

    let decoded: JwtPayload;
    try {
      decoded = jwt.verify(token, jwtSecret) as JwtPayload;
    } catch {
      throw new ApiError(401, "Invalid or expired token");
    }

    const user = await User.findById(decoded.userId).select("_id");
    if (!user) throw new ApiError(401, "User no longer exists");

    req.userId = user._id;
    next();
  } catch (err) {
    next(err);
  }
};

import type { Request, Response, NextFunction } from "express";
import { ApiError } from "../utils/api-error.js";

interface MongooseCastError extends Error { name: "CastError"; path: string; }
interface MongooseValidationError extends Error { name: "ValidationError"; errors: Record<string, { message: string }>; }
interface MongooseDuplicateKeyError extends Error { code: number; }

const isCastError = (e: unknown): e is MongooseCastError =>
  e instanceof Error && e.name === "CastError";
const isValidationError = (e: unknown): e is MongooseValidationError =>
  e instanceof Error && e.name === "ValidationError";
const isDuplicateKeyError = (e: unknown): e is MongooseDuplicateKeyError =>
  e instanceof Error && "code" in e && (e as MongooseDuplicateKeyError).code === 11000;

/**
 * Global error handler — last middleware in the chain.
 * Handles ApiError, Mongoose errors, and unknown errors gracefully.
 * The app NEVER crashes from an unhandled route/controller error.
 */
export const globalErrorHandler = (
  err: unknown,
  _req: Request,
  res: Response,
  _next: NextFunction
): void => {
  if (err instanceof ApiError) {
    res.status(err.statusCode).json({
      success: false,
      statusCode: err.statusCode,
      message: err.message,
      errors: err.errors.length > 0 ? err.errors : undefined,
      data: null,
    });
    return;
  }

  if (isCastError(err)) {
    res.status(400).json({ success: false, statusCode: 400, message: `Invalid value for field: ${err.path}`, data: null });
    return;
  }

  if (isValidationError(err)) {
    const messages = Object.values(err.errors).map((e) => e.message);
    res.status(400).json({ success: false, statusCode: 400, message: "Validation failed", errors: messages, data: null });
    return;
  }

  if (isDuplicateKeyError(err)) {
    res.status(409).json({ success: false, statusCode: 409, message: "A record with that value already exists", data: null });
    return;
  }

  console.error("[Unhandled Error]", err);
  res.status(500).json({ success: false, statusCode: 500, message: "Internal Server Error", data: null });
};

export const buildResumeTextForAI = (resume: ResumeContent): string => {
  const lines: string[] = [];
  const pi = resume.personal_info ?? {};

  if (pi.full_name)  lines.push(`Name: ${pi.full_name}`);
  if (pi.profession) lines.push(`Profession: ${pi.profession}`);
  if (pi.location)   lines.push(`Location: ${pi.location}`);

  if (resume.professional_summary) {
    lines.push("\nPROFESSIONAL SUMMARY");
    lines.push(resume.professional_summary);
  }
  if (resume.skills?.length) {
    lines.push("\nSKILLS");
    lines.push(resume.skills.join(", "));
  }
  if (resume.experience?.length) {
    lines.push("\nEXPERIENCE");
    resume.experience.forEach((exp) => {
      const end = exp.is_current ? "Present" : (exp.end_date ?? "");
      lines.push(`${exp.position} at ${exp.company} (${exp.start_date ?? ""} – ${end})`);
      if (exp.description) lines.push(exp.description);
    });
  }
  if (resume.education?.length) {
    lines.push("\nEDUCATION");
    resume.education.forEach((edu) => {
      lines.push(`${edu.degree} in ${edu.field} — ${edu.institution} (${edu.graduation_date ?? ""})`);
    });
  }
  if (resume.project?.length) {
    lines.push("\nPROJECTS");
    resume.project.forEach((proj) => {
      lines.push(`${proj.name} (${proj.type}): ${proj.description}`);
    });
  }

  return lines.join("\n");
};

import type { IResume } from "../models/resume.model.js";
import type { ResumeContent } from "../types/index.js";

export const VERSION_CAP = 10;

/**
 * Builds a clean content snapshot from a resume document.
 * Strips non-content fields (_id, userId, views, downloads, versions, timestamps)
 * so the snapshot only contains editable resume fields.
 */
export const buildSnapshot = (resume: IResume): ResumeContent => {
  const obj = resume.toObject() as Record<string, unknown>;
  const omitKeys = new Set([
    "_id", "userId", "createdAt", "updatedAt",
    "views", "downloads", "versions", "__v",
  ]);
  const snapshot: Record<string, unknown> = {};
  for (const [key, value] of Object.entries(obj)) {
    if (!omitKeys.has(key)) snapshot[key] = value;
  }
  return snapshot as ResumeContent;
};

/**
 * Appends a version snapshot of the current resume state to its versions array,
 * capping at VERSION_CAP by dropping the oldest entry.
 *
 * @param resume  - The Mongoose resume document to snapshot.
 * @param label   - Optional custom label. Auto-generated if omitted.
 * @param suffix  - Optional suffix appended to the auto-generated label (e.g. "(before restore)").
 */
export const pushVersion = (
  resume: IResume,
  label?: string,
  suffix?: string
): IResume["versions"] => {
  const snapshot   = buildSnapshot(resume);
  const versionNum = (resume.versions?.length ?? 0) + 1;
  const dateStr    = new Date().toLocaleDateString("en-US", {
    month: "short",
    day:   "numeric",
  });
  const baseLabel   = label ?? `Version ${versionNum} — ${dateStr}`;
  const finalLabel  = suffix ? `${baseLabel} ${suffix}` : baseLabel;

  return [
    ...(resume.versions ?? []).slice(-(VERSION_CAP - 1)),
    { savedAt: new Date(), label: finalLabel, snapshot },
  ];
};

export class ApiResponse<T = unknown> {
  public readonly statusCode: number;
  public readonly data: T | null;
  public readonly message: string;
  public readonly success: boolean;

  constructor(statusCode: number, data: T | null, message = "Success") {
    this.statusCode = statusCode;
    this.data = data;
    this.message = message;
    this.success = statusCode >= 200 && statusCode < 300;
  }

  send(res: Response): Response<ApiResponseShape<T>> {
    return res.status(this.statusCode).json({
      success: this.success,
      statusCode: this.statusCode,
      message: this.message,
      data: this.data,
    });
  }
}

import type { Request, Response, NextFunction, RequestHandler } from "express";

type AsyncRequestHandler<
  P = Record<string, string>,
  ResBody = unknown,
  ReqBody = unknown,
  ReqQuery = Record<string, string>,
> = (
  req: Request<P, ResBody, ReqBody, ReqQuery>,
  res: Response<ResBody>,
  next: NextFunction
) => Promise<void | Response>;

/**
 * Wraps an async route handler and forwards any thrown errors to Express's
 * next(err) so the global error handler can process them.
 *
 * This means controllers never need try/catch — they just throw ApiError.
 */
export const asyncHandler = <
  P = Record<string, string>,
  ResBody = unknown,
  ReqBody = unknown,
  ReqQuery = Record<string, string>,
>(
  fn: AsyncRequestHandler<P, ResBody, ReqBody, ReqQuery>
): RequestHandler<P, ResBody, ReqBody, ReqQuery> => {
  return (req, res, next) => {
    Promise.resolve(fn(req, res, next)).catch(next);
  };
};

import jwt from "jsonwebtoken";
import type { Types } from "mongoose";

const getSecret = (key: string): string => {
  const secret = process.env[key];
  if (!secret) throw new Error(`Missing env var: ${key}`);
  return secret;
};

export const generateAccessToken = (userId: Types.ObjectId): string => {
  return jwt.sign({ userId: userId.toString() }, getSecret("JWT_SECRET"), {
    expiresIn: "15m",
  });
};

export const generateRefreshToken = (userId: Types.ObjectId): string => {
  return jwt.sign(
    { userId: userId.toString() },
    getSecret("JWT_REFRESH_SECRET"),
    { expiresIn: "7d" }
  );
};

import validator from "validator";
import type { Request } from "express";
import { ApiError } from "./api-error.js";

export interface SignUpBody {
  firstName?: string;
  lastName?: string;
  emailId?: string;
  password?: string;
}

export const validateSignUpData = (req: Request<unknown, unknown, SignUpBody>): void => {
  const { firstName, lastName, emailId, password } = req.body;

  if (!firstName?.trim() || !lastName?.trim()) {
    throw new ApiError(400, "First name and last name are required");
  }

  if (!validator.isEmail(emailId ?? "")) {
    throw new ApiError(400, "Please provide a valid email address");
  }

  if (!password || password.length < 8) {
    throw new ApiError(400, "Password must be at least 8 characters");
  }

  if (!/[A-Z]/.test(password)) {
    throw new ApiError(400, "Password must contain at least one uppercase letter");
  }

  if (!/\d/.test(password)) {
    throw new ApiError(400, "Password must contain at least one number");
  }
};

export const validatePasswordStrength = (password: string): void => {
  if (password.length < 8) throw new ApiError(400, "Password must be at least 8 characters");
  if (!/[A-Z]/.test(password)) throw new ApiError(400, "Password must contain at least one uppercase letter");
  if (!/\d/.test(password)) throw new ApiError(400, "Password must contain at least one number");
};
