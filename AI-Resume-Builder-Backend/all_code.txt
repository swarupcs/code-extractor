import passport from 'passport';
import { Strategy as GoogleStrategy } from 'passport-google-oauth20';
import User from '../models/user.model.js';

// Strategy is registered inside initializePassport() — NOT at module load time.
//
// Root cause of the original error:
//   ES module imports are hoisted and executed before any code in the importing
//   file runs. So when app.js does `import './config/passport.js'`, the module
//   executes immediately — before dotenv.config() has a chance to populate
//   process.env. Result: process.env.GOOGLE_CLIENT_ID is undefined, and
//   passport-google-oauth20 throws "OAuth2Strategy requires a clientID option".
//
// Fix: export an initializePassport() function. app.js calls dotenv.config()
// first, then calls initializePassport() — by that point all env vars are set.

export const initializePassport = () => {
  if (!process.env.GOOGLE_CLIENT_ID || !process.env.GOOGLE_CLIENT_SECRET) {
    console.warn(
      '[Passport] GOOGLE_CLIENT_ID or GOOGLE_CLIENT_SECRET not set — ' +
      'Google OAuth will not be available.'
    );
    return;
  }

  passport.use(
    new GoogleStrategy(
      {
        clientID:     process.env.GOOGLE_CLIENT_ID,
        clientSecret: process.env.GOOGLE_CLIENT_SECRET,
        callbackURL:  process.env.GOOGLE_CALLBACK_URL,
      },
      async (_accessToken, _refreshToken, profile, done) => {
        try {
          const email     = profile.emails?.[0]?.value?.toLowerCase();
          const firstName = profile.name?.givenName  || profile.displayName || 'User';
          const lastName  = profile.name?.familyName || '';

          if (!email) {
            return done(new Error('No email returned from Google'), null);
          }

          // 1. Returning user — find by googleId
          let user = await User.findOne({ googleId: profile.id });

          if (!user) {
            // 2. Existing email/password account — link it
            user = await User.findOne({ emailId: email });

            if (user) {
              user.googleId   = profile.id;
              user.isVerified = true;
              await user.save();
            } else {
              // 3. Brand new user — create account
              user = await User.create({
                firstName,
                lastName,
                emailId:    email,
                googleId:   profile.id,
                isVerified: true,
                password:   null,
              });
            }
          }

          return done(null, user);
        } catch (error) {
          return done(error, null);
        }
      }
    )
  );

  // No sessions — JWTs handle auth state
  passport.serializeUser((user, done) => done(null, user._id));
  passport.deserializeUser(async (id, done) => {
    try {
      const user = await User.findById(id).select('-password -refreshToken');
      done(null, user);
    } catch (err) {
      done(err, null);
    }
  });

  console.log('[Passport] Google OAuth strategy registered.');
};

export default passport;

// ── ATS Keyword Analysis ──────────────────────────────────────────────────────
//
// Compares the current resume (frontend state, not DB — includes unsaved changes)
// against a pasted job description and returns:
//   score           — 0-100 ATS compatibility score
//   summary         — one-line assessment
//   matchedKeywords — keywords present in both (green)
//   missingKeywords — important JD keywords absent from resume (red)
//   suggestions     — 3-5 specific, actionable improvements
//
// Resume text is built server-side from structured fields to prevent
// prompt injection through resume content fields.

export const analyzeATS = asyncHandler(async (req, res) => {
  const { resumeData, jobDescription } = req.body;

  if (!resumeData || !jobDescription) {
    throw new ApiError(400, 'Both resume data and job description are required');
  }
  if (jobDescription.trim().length < 50) {
    throw new ApiError(400, 'Please paste the full job description (at least 50 characters)');
  }

  // Parse resumeData if sent as string
  let parsed;
  try {
    parsed = typeof resumeData === 'string' ? JSON.parse(resumeData) : resumeData;
  } catch {
    throw new ApiError(400, 'Invalid resume data format');
  }

  const resumeText = buildResumeText(parsed);

  const systemPrompt = `You are an expert ATS (Applicant Tracking System) analyst and resume coach.
Compare a resume against a job description and return a structured JSON analysis.
Always respond with valid JSON only — no markdown, no explanation outside the JSON.`;

  const userPrompt = `Analyse this resume against the job description and return a JSON object.

RESUME:
${resumeText}

JOB DESCRIPTION:
${jobDescription}

Return ONLY this exact JSON structure:
{
  "score": <integer 0-100>,
  "summary": "<one sentence overall assessment>",
  "matchedKeywords": ["<keyword>"],
  "missingKeywords": ["<keyword>"],
  "suggestions": [
    "<specific actionable suggestion 1>",
    "<specific actionable suggestion 2>",
    "<specific actionable suggestion 3>"
  ]
}

Scoring guide:
- 80-100: Excellent — most keywords present, strong alignment
- 60-79:  Good — some important keywords missing
- 40-59:  Moderate — significant gaps
- 0-39:   Poor — major gaps or wrong field

Rules:
- matchedKeywords: skills, tools, qualifications found in BOTH resume and JD
- missingKeywords: the most important JD keywords completely absent from the resume (max 15)
- suggestions: be specific — name exact skills or sections, not generic advice. Under 20 words each.
- Return 3 to 5 suggestions`;

  const response = await ai.chat.completions.create({
    model: process.env.OPENAI_MODEL,
    messages: [
      { role: 'system', content: systemPrompt },
      { role: 'user',   content: userPrompt },
    ],
    response_format: { type: 'json_object' },
  });

  const raw    = response.choices[0].message.content;
  const clean  = raw.replace(/```json\n?|```/g, '').trim();
  const result = JSON.parse(clean);

  // Sanitise the response to guarantee shape
  const analysis = {
    score:           Math.min(100, Math.max(0, parseInt(result.score, 10) || 0)),
    summary:         String(result.summary  || ''),
    matchedKeywords: Array.isArray(result.matchedKeywords) ? result.matchedKeywords.slice(0, 30) : [],
    missingKeywords: Array.isArray(result.missingKeywords) ? result.missingKeywords.slice(0, 15) : [],
    suggestions:     Array.isArray(result.suggestions)     ? result.suggestions.slice(0, 5)      : [],
  };

  return new ApiResponse(200, { analysis }, 'ATS analysis complete').send(res);
});


// Build a readable flat text from structured resume JSON — server-side only
function buildResumeText(resume) {
  const lines = [];
  const pi    = resume.personal_info ?? {};

  if (pi.full_name)  lines.push(`Name: ${pi.full_name}`);
  if (pi.profession) lines.push(`Profession: ${pi.profession}`);
  if (pi.location)   lines.push(`Location: ${pi.location}`);

  if (resume.professional_summary) {
    lines.push('\nPROFESSIONAL SUMMARY');
    lines.push(resume.professional_summary);
  }

  if (resume.skills?.length) {
    lines.push('\nSKILLS');
    lines.push(resume.skills.join(', '));
  }

  if (resume.experience?.length) {
    lines.push('\nEXPERIENCE');
    resume.experience.forEach((exp) => {
      lines.push(
        `${exp.position} at ${exp.company} (${exp.start_date ?? ''} – ${exp.is_current ? 'Present' : (exp.end_date ?? '')})`
      );
      if (exp.description) lines.push(exp.description);
    });
  }

  if (resume.education?.length) {
    lines.push('\nEDUCATION');
    resume.education.forEach((edu) => {
      lines.push(`${edu.degree} in ${edu.field} — ${edu.institution} (${edu.graduation_date ?? ''})`);
    });
  }

  if (resume.project?.length) {
    lines.push('\nPROJECTS');
    resume.project.forEach((proj) => {
      lines.push(`${proj.name} (${proj.type}): ${proj.description}`);
    });
  }

  return lines.join('\n');
}

import bcrypt from 'bcryptjs';
import crypto from 'crypto';
import jwt from 'jsonwebtoken';
import User from '../models/user.model.js';
import { ApiError } from '../utils/api-error.js';
import { ApiResponse } from '../utils/api-response.js';
import { asyncHandler } from '../utils/async-handler.js';
import {
  setAuthCookie, clearAuthCookie,
  setRefreshCookie, clearRefreshCookie,
} from '../utils/cookies.js';
import {
  generateAccessToken,
  generateRefreshToken,
} from '../utils/generateToken.js';
import { validateSignUpData } from '../utils/validation.js';
import {
  sendVerificationEmail,
  sendPasswordResetEmail,
} from '../utils/email.js';


// ── Shared helper ─────────────────────────────────────────────────────────────
const issueTokens = async (res, user) => {
  const accessToken  = generateAccessToken(user._id);
  const refreshToken = generateRefreshToken(user._id);

  user.refreshToken = await bcrypt.hash(refreshToken, 8);
  await user.save();

  setAuthCookie(res, accessToken);
  setRefreshCookie(res, refreshToken);

  return { accessToken, refreshToken };
};

// Generate a cryptographically random hex token and its SHA-256 hash.
// We store the hash in the DB and send the raw token in the email so a leaked
// DB doesn't let an attacker use stored tokens.
const generateSecureToken = () => {
  const raw  = crypto.randomBytes(32).toString('hex');
  const hash = crypto.createHash('sha256').update(raw).digest('hex');
  return { raw, hash };
};


// ── Signup ────────────────────────────────────────────────────────────────────
export const signupUser = asyncHandler(async (req, res) => {
  const { firstName, lastName, emailId, password } = req.body;

  validateSignUpData(req);

  const existingUser = await User.findOne({ emailId });
  if (existingUser) throw new ApiError(409, 'User already exists with this email');

  const hashedPassword = await bcrypt.hash(password, 10);

  // Generate email verification token
  const { raw: verifyToken, hash: verifyHash } = generateSecureToken();

  const user = await User.create({
    firstName,
    lastName,
    emailId,
    password:                 hashedPassword,
    isVerified:               false,
    emailVerificationToken:   verifyHash,
    emailVerificationExpires: new Date(Date.now() + 24 * 60 * 60 * 1000), // 24h
  });

  // Send verification email — fire-and-forget, don't block signup on email failure
  sendVerificationEmail({
    to:        emailId,
    firstName,
    token:     verifyToken,
  }).catch((err) => console.error('Verification email failed:', err));

  // Issue tokens so the user is logged in immediately after signup
  await issueTokens(res, user);

  user.password               = undefined;
  user.refreshToken           = undefined;
  user.emailVerificationToken = undefined;

  return new ApiResponse(
    201,
    { user, requiresVerification: true },
    'Account created. Please check your email to verify your account.'
  ).send(res);
});


// ── Login ─────────────────────────────────────────────────────────────────────
export const loginUser = asyncHandler(async (req, res) => {
  const { emailId, password } = req.body;

  if (!emailId || !password) throw new ApiError(400, 'Email and password are required');

  const user = await User.findOne({ emailId });
  if (!user) throw new ApiError(401, 'Invalid email or password');

  // Google-only users have no password
  if (!user.password) {
    throw new ApiError(400, 'This account uses Google sign-in. Please continue with Google.');
  }

  const isPasswordValid = user.comparePassword(password);
  if (!isPasswordValid) throw new ApiError(401, 'Invalid email or password');

  // Block login for unverified accounts
  // if (!user.isVerified) {
  //   throw new ApiError(
  //     403,
  //     'Please verify your email address before logging in. Check your inbox for the verification link.'
  //   );
  // }

  await issueTokens(res, user);

  user.password     = undefined;
  user.refreshToken = undefined;

  return new ApiResponse(200, { user }, 'Login successful').send(res);
});

export const logoutUser = asyncHandler(async (req, res) => {

// ── Refresh tokens ────────────────────────────────────────────────────────────
export const refreshTokens = asyncHandler(async (req, res) => {
  const incomingRefresh = req.cookies?.refreshToken;
  if (!incomingRefresh) throw new ApiError(401, 'No refresh token provided');

  let decoded;
  try {
    decoded = jwt.verify(incomingRefresh, process.env.JWT_REFRESH_SECRET);
  } catch {
    throw new ApiError(401, 'Invalid or expired refresh token');
  }

  const user = await User.findById(decoded.userId);
  if (!user || !user.refreshToken) {
    throw new ApiError(401, 'Session expired — please log in again');
  }

  const isValid = await bcrypt.compare(incomingRefresh, user.refreshToken);
  if (!isValid) throw new ApiError(401, 'Refresh token reuse detected — please log in again');

  await issueTokens(res, user);

  user.password     = undefined;
  user.refreshToken = undefined;

  return new ApiResponse(200, { user }, 'Tokens refreshed successfully').send(res);
});


// ── Verify email ──────────────────────────────────────────────────────────────
export const verifyEmail = asyncHandler(async (req, res) => {
  const { token } = req.params;
  if (!token) throw new ApiError(400, 'Verification token is required');

  const hash = crypto.createHash('sha256').update(token).digest('hex');

  const user = await User.findOne({
    emailVerificationToken:   hash,
    emailVerificationExpires: { $gt: Date.now() },
  });

  if (!user) {
    throw new ApiError(400, 'Verification link is invalid or has expired');
  }

  user.isVerified               = true;
  user.emailVerificationToken   = undefined;
  user.emailVerificationExpires = undefined;
  await user.save();

  return new ApiResponse(200, null, 'Email verified successfully').send(res);
});


// ── Resend verification email ─────────────────────────────────────────────────
export const resendVerification = asyncHandler(async (req, res) => {
  const { emailId } = req.body;
  if (!emailId) throw new ApiError(400, 'Email is required');

  const user = await User.findOne({ emailId });

  // Always return success to prevent email enumeration attacks
  if (!user || user.isVerified) {
    return new ApiResponse(
      200,
      null,
      'If your account exists and is unverified, a new verification email has been sent.'
    ).send(res);
  }

  const { raw: verifyToken, hash: verifyHash } = generateSecureToken();

  user.emailVerificationToken   = verifyHash;
  user.emailVerificationExpires = new Date(Date.now() + 24 * 60 * 60 * 1000);
  await user.save();

  sendVerificationEmail({
    to:        user.emailId,
    firstName: user.firstName,
    token:     verifyToken,
  }).catch((err) => console.error('Resend verification email failed:', err));

  return new ApiResponse(

// ── Reset password ────────────────────────────────────────────────────────────
export const resetPassword = asyncHandler(async (req, res) => {
  const { token }    = req.params;
  const { password } = req.body;

  if (!token)    throw new ApiError(400, 'Reset token is required');
  if (!password) throw new ApiError(400, 'New password is required');
  if (password.length < 8) throw new ApiError(400, 'Password must be at least 8 characters');
  if (!/[A-Z]/.test(password)) throw new ApiError(400, 'Password must contain at least one uppercase letter');
  if (!/\d/.test(password))    throw new ApiError(400, 'Password must contain at least one number');

  const hash = crypto.createHash('sha256').update(token).digest('hex');

  const user = await User.findOne({
    resetPasswordToken:   hash,
    resetPasswordExpires: { $gt: Date.now() },
  });

  if (!user) {
    throw new ApiError(400, 'Password reset link is invalid or has expired');
  }

  user.password             = await bcrypt.hash(password, 10);
  user.resetPasswordToken   = undefined;
  user.resetPasswordExpires = undefined;
  // Invalidate all existing sessions after password reset
  user.refreshToken         = null;
  await user.save();

  clearAuthCookie(res);
  clearRefreshCookie(res);

  return new ApiResponse(200, null, 'Password reset successful. Please log in with your new password.').send(res);
});

// Phase 5 — Feature 5: updateResume pushes a version snapshot before saving.
export const updateResume = asyncHandler(async (req, res) => {
  const userId = req.userId;
  const { resumeId, resumeData, removeBackground } = req.body;
  const image = req.file;

  if (!resumeId) throw new ApiError(400, 'Resume ID is required');
  if (!resumeData) throw new ApiError(400, 'Resume data is required');

  let parsed;
  try {
    parsed = typeof resumeData === 'string' ? JSON.parse(resumeData) : resumeData;
  } catch {
    throw new ApiError(400, 'Invalid resume data format');
  }

  const allowedUpdate = {
    title:                parsed.title,
    personal_info:        parsed.personal_info        ?? {},
    professional_summary: parsed.professional_summary ?? '',
    experience:           parsed.experience           ?? [],
    education:            parsed.education            ?? [],
    project:              parsed.project              ?? [],
    skills:               parsed.skills               ?? [],
    template:             parsed.template             ?? 'classic',
    accent_color:         parsed.accent_color         ?? '#3B82F6',
    ...(typeof parsed.public === 'boolean' && { isPublic: parsed.public }),
  };

  if (image) {
    const shouldRemoveBg = removeBackground === true || removeBackground === 'true';
    try {
      const imageBufferData = fs.createReadStream(image.path);
      const response = await imagekit.files.upload({
        file:     imageBufferData,
        fileName: `resume-${resumeId}.png`,
        folder:   'user-resumes',
        transformation: {
          pre: `w-300,h-300,fo-face,z-0.75${shouldRemoveBg ? ',e-bgremove' : ''}`,
        },
      });
      allowedUpdate.personal_info = { ...allowedUpdate.personal_info, image: response.url };
    } finally {
      if (fs.existsSync(image.path)) fs.unlinkSync(image.path);
    }
  }

  // Fetch current state to snapshot BEFORE overwriting
  const existing = await Resume.findOne({ _id: resumeId, userId });
  if (!existing) throw new ApiError(404, 'Resume not found or access denied');

  const snapshot   = buildSnapshot(existing);
  const versionNum = (existing.versions?.length ?? 0) + 1;
  const label      = `Version ${versionNum} — ${new Date().toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}`;

  // Push snapshot, cap at VERSION_CAP by slicing from the end
  const updatedVersions = [
    ...(existing.versions ?? []).slice(-(VERSION_CAP - 1)),
    { savedAt: new Date(), label, snapshot },
  ];

  const resume = await Resume.findOneAndUpdate(
    { _id: resumeId, userId },
    { ...allowedUpdate, versions: updatedVersions },
    { new: true }
  );

  return new ApiResponse(200, { resume }, 'Resume saved successfully').send(res);
});

export const restoreVersion = asyncHandler(async (req, res) => {
  const userId = req.userId;
  const { id, versionIndex } = req.params;

  const resume = await Resume.findOne({ _id: id, userId });
  if (!resume) throw new ApiError(404, 'Resume not found or access denied');

  const idx = parseInt(versionIndex, 10);
  if (isNaN(idx) || idx < 0 || idx >= resume.versions.length) {
    throw new ApiError(400, 'Invalid version index');
  }

  const { snapshot } = resume.versions[idx];

  // Snapshot before restoring so the current state isn't lost
  const currentSnapshot = buildSnapshot(resume);
  const versionNum      = (resume.versions.length) + 1;
  const label           = `Version ${versionNum} — ${new Date().toLocaleDateString('en-US', { month: 'short', day: 'numeric' })} (before restore)`;

  const updatedVersions = [
    ...resume.versions.slice(-(VERSION_CAP - 1)),
    { savedAt: new Date(), label, snapshot: currentSnapshot },
  ];

  const restored = await Resume.findOneAndUpdate(
    { _id: id, userId },
    { ...snapshot, versions: updatedVersions },
    { new: true }
  );

  return new ApiResponse(200, { resume: restored }, 'Version restored successfully').send(res);
});

import jwt from 'jsonwebtoken';
import User from '../models/user.model.js';
import { asyncHandler } from '../utils/async-handler.js';
import { ApiError } from '../utils/api-error.js';


/**
 * Middleware to authenticate user using JWT token stored in cookies
 */
export const authMiddleware = asyncHandler(async (req, res, next) => {
  // 1️⃣ Get token from cookies
  const token = req.cookies?.token;

  if (!token) {
    throw new ApiError(401, 'Please login first');
  }

  // 2️⃣ Verify token
  let decoded;
  try {
    decoded = jwt.verify(token, process.env.JWT_SECRET);
  } catch (error) {
    throw new ApiError(401, 'Invalid or expired token');
  }

  // 3️⃣ Check if user still exists
  const user = await User.findById(decoded.userId).select('_id');
  if (!user) {
    throw new ApiError(401, 'User no longer exists');
  }

  // 4️⃣ Attach userId to request (IMPORTANT)
  req.userId = user._id;

  next();
});

class ApiError extends Error {
  constructor(
    statusCode,
    message = 'Something went wrong',
    errors = [],
    stack = ''
  ) {
    super(message);
    this.statusCode = statusCode;
    this.data = null;
    this.name = this.constructor.name;
    this.message = message;
    this.success = false;
    this.errors = errors;

    if (stack) {
      this.stack = stack;
    } else {
      Error.captureStackTrace(this, this.constructor); // Error.captureStackTrace(this, this.constructor) is called to generate a stack trace automatically.
    }
  }
}

export { ApiError };

class ApiResponse {
  constructor(statusCode, data, message = 'Success') {
    this.statusCode = statusCode;
    this.data = data;
    this.message = message;
    this.success = statusCode >= 200 && statusCode < 300; // success is true for status codes in the range of 200-299
  }

  send(res) {
    return res.status(this.statusCode).json({
      success: this.success,
      statusCode: this.statusCode,
      message: this.message,
      data: this.data,
    });
  }
}

export { ApiResponse };

import jwt from 'jsonwebtoken';

// Phase 5 — Feature 2: Refresh token rotation.
// Two tokens:
//   Access token  — short-lived (15 min), sent as httpOnly cookie on every request.
//   Refresh token — long-lived (7 days), stored in a separate httpOnly cookie.
//                   Used only to obtain a new access token when it expires.
//
// Why separate cookies rather than Authorization headers?
//   The app already uses httpOnly cookies for auth — keeping both tokens in
//   httpOnly cookies means XSS attacks can never read either token.

export const generateAccessToken = (userId) => {
  return jwt.sign({ userId }, process.env.JWT_SECRET, { expiresIn: '15m' });
};

export const generateRefreshToken = (userId) => {
  return jwt.sign({ userId }, process.env.JWT_REFRESH_SECRET, { expiresIn: '7d' });
};

// Keep the old export name as an alias for any code that hasn't migrated yet
export const generateToken = generateAccessToken;

import validator from 'validator';
import { ApiError } from './api-error.js';

/**
 * Validate signup request body.
 * FIX: Was imported but commented out in auth.controller.js — never actually ran.
 * Now throws ApiError (not plain Error) so the global error handler returns
 * a structured JSON response instead of crashing with a 500.
 */
export const validateSignUpData = (req) => {
  const { firstName, lastName, emailId, password } = req.body;

  if (!firstName?.trim() || !lastName?.trim()) {
    throw new ApiError(400, 'First name and last name are required');
  }

  if (!validator.isEmail(emailId || '')) {
    throw new ApiError(400, 'Please provide a valid email address');
  }

  if (!password || password.length < 8) {
    throw new ApiError(400, 'Password must be at least 8 characters');
  }

  if (!/[A-Z]/.test(password)) {
    throw new ApiError(
      400,
      'Password must contain at least one uppercase letter',
    );
  }

  if (!/\d/.test(password)) {
    throw new ApiError(400, 'Password must contain at least one number');
  }
};

 */
export const validateEditProfileData = (req) => {

    'emailId',

    'gender',

  ];

  return Object.keys(req.body).every((field) =>
    allowedEditFields.includes(field),
  );
};
