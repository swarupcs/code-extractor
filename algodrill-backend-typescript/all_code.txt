// src/app.ts
import 'express-async-errors'; // must be first import
import type { Application, Request, Response } from 'express';
import express from 'express';
import cors from 'cors';
import cookieParser from 'cookie-parser';
import helmet from 'helmet';
import rateLimit from 'express-rate-limit';
import dotenv from 'dotenv';

import { globalErrorHandler, notFoundHandler } from './middleware/errorHandler';
import { logger } from './utils/logger';
import { db } from './libs/db';

import authRoutes from './routes/auth.routes';
import problemRoutes from './routes/problem.routes';
import submissionRoutes from './routes/submission.routes';
import runCodeRoutes from './routes/runCode.routes';
import sheetsRoutes from './routes/sheets.routes';
import userStatsRoutes from './routes/userStats.routes';
import discussionsRoutes from './routes/discussions.routes';
import leaderboardRoutes from './routes/leaderboard.routes';
import roadmapRoutes from './routes/roadmaps.routes';
import adminRoutes from './routes/admin.routes';

dotenv.config();

// ─── Validate required env vars at startup ────────────────────────────────────
const REQUIRED_ENV = [
  'DATABASE_URL',
  'JWT_SECRET',
  'ALLOWED_ORIGINS',
  'SESSION_SECRET',
  'GOOGLE_CLIENT_ID',
  'GOOGLE_CLIENT_SECRET',
  'GOOGLE_CALLBACK_URL',
];
for (const key of REQUIRED_ENV) {
  if (!process.env[key]) {
    logger.error(`❌  Missing required environment variable: ${key}`);
    process.exit(1);
  }
}

const app: Application = express();

app.set('trust proxy', 1);

// ─── Security ─────────────────────────────────────────────────────────────────
app.use(helmet());

const allowedOrigins = (process.env.ALLOWED_ORIGINS as string)
  .split(',')
  .map((o) => o.trim().replace(/\r/g, ''));

app.use(
  cors({
    origin: (origin, callback) => {
      if (!origin) return callback(null, true);
      if (allowedOrigins.includes(origin)) return callback(null, true);
      return callback(new Error(`CORS: Origin ${origin} not allowed`));
    },
    credentials: true,
  }),
);

// ─── Rate limiting ────────────────────────────────────────────────────────────
const globalLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 200,
  standardHeaders: true,
  legacyHeaders: false,
  message: { success: false, message: 'Too many requests, please try again later.' },
});

const authLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 30,
  standardHeaders: true,
  legacyHeaders: false,
  message: { success: false, message: 'Too many auth attempts, please try again later.' },
});

app.use(globalLimiter);

// ─── Body parsing ─────────────────────────────────────────────────────────────
app.use(express.json({ limit: '1mb' }));
app.use(express.urlencoded({ extended: true }));
app.use(cookieParser());

// ─── Health check ─────────────────────────────────────────────────────────────
app.get('/', (_req: Request, res: Response) => {
  res.json({ success: true, message: 'AlgoDrill API is running 🚀' });
});

app.get('/ping', (_req: Request, res: Response) => {
  res.json({ success: true, message: 'pong', timestamp: new Date().toISOString() });
});

// ─── Routes ───────────────────────────────────────────────────────────────────
app.use('/api/v1/auth', authLimiter, authRoutes);
app.use('/api/v1/problems', problemRoutes);
app.use('/api/v1/code', runCodeRoutes);
app.use('/api/v1/submissions', submissionRoutes);
app.use('/api/v1/sheets', sheetsRoutes);
app.use('/api/v1/user-stats', userStatsRoutes);
app.use('/api/v1/discussions', discussionsRoutes);
app.use('/api/v1/leaderboard', leaderboardRoutes);
app.use('/api/v1/roadmaps', roadmapRoutes);
app.use('/api/v1/admin', adminRoutes);

// ─── Error handling ───────────────────────────────────────────────────────────
app.use(notFoundHandler);
app.use(globalErrorHandler);

// ─── Graceful shutdown ────────────────────────────────────────────────────────
async function shutdown(signal: string): Promise<void> {
  logger.info(`\n${signal} received — shutting down gracefully...`);
  try {
    await db.$disconnect();
    logger.info('✅  Database disconnected');
  } catch (err) {
    logger.error('Error during DB disconnect:', err);
  }
  process.exit(0);
}

process.on('SIGTERM', () => void shutdown('SIGTERM'));
process.on('SIGINT', () => void shutdown('SIGINT'));

process.on('uncaughtException', (err) => {
  logger.error('Uncaught Exception:', err);
  process.exit(1);
});

process.on('unhandledRejection', (reason) => {
  logger.error('Unhandled Rejection:', reason);
  process.exit(1);
});

// ─── Start ────────────────────────────────────────────────────────────────────
const PORT = parseInt(process.env.PORT ?? '8080', 10);
const ENV = process.env.NODE_ENV ?? 'development';

async function start(): Promise<void> {
  // 1. Verify DB connection before accepting traffic
  logger.info('⏳  Connecting to database...');
  try {
    await db.$connect();
    logger.info('✅  Database connected');
  } catch (err) {
    logger.error('❌  Failed to connect to database:', err);
    process.exit(1);
  }

  // 2. Start HTTP server
  app.listen(PORT, () => {
    logger.info('');
    logger.info('┌─────────────────────────────────────────┐');
    logger.info(`│   🚀  AlgoDrill API                      │`);
    logger.info(`│   ➜   http://localhost:${PORT}             │`);
    logger.info(`│   ENV  ${ENV.padEnd(33)}│`);
    logger.info('└─────────────────────────────────────────┘');
    logger.info('');
  });
}

void start();

export default app;

// src/config/passport.ts
import passport from 'passport';
import {
  Strategy as GoogleStrategy,
  type StrategyOptions,
  type Profile,
} from 'passport-google-oauth20';
import { linkGoogleAccount } from '../services/auth.service';
import { logger } from '../utils/logger';

// ─── Google OAuth Strategy ────────────────────────────────────────────────────

passport.use(
  new GoogleStrategy(
    {
      clientID: process.env.GOOGLE_CLIENT_ID as string,
      clientSecret: process.env.GOOGLE_CLIENT_SECRET as string,
      callbackURL: process.env.GOOGLE_CALLBACK_URL as string,
    } satisfies StrategyOptions,
    (
      _accessToken: string,
      _refreshToken: string,
      profile: Profile, // ← top-level import type
      done: (err: Error | null, user?: Express.User | false) => void,
    ) => {
      const email = profile.emails?.[0]?.value;
      if (!email) return done(new Error('No email provided by Google'), undefined);

      linkGoogleAccount(
        // ← no async/await, use .then/.catch
        profile.id,
        email,
        profile.displayName,
        profile.photos?.[0]?.value,
      )
        .then((user) => done(null, user as Express.User))
        .catch((err: Error) => {
          logger.error('Passport Google strategy error:', err);
          done(err, undefined);
        });
    },
  ),
);

// ─── Session Serialization ────────────────────────────────────────────────────

passport.serializeUser((user, done) => {
  done(null, (user as { id: string }).id);
});

passport.deserializeUser((id: string, done) => {
  // ← non-async, use .then/.catch
  import('../libs/db')
    .then(({ db }) => db.user.findUnique({ where: { id } }))
    .then((user) => done(null, user as Express.User | null))
    .catch((err: unknown) => done(err, null));
});

export default passport;

// src/controllers/admin.controller.ts
import type { Request, Response, RequestHandler } from 'express';
import { ApiResponse } from '../utils/ApiResponse';
import { asyncHandler } from '../utils/asyncHandler';
import { AppError } from '../utils/AppError';
import { getAdminDashboard, getAdminUserDetail, getAdminUsers, updateAdminUserRole } from '../services/admin.service';

/**
 * GET /api/v1/admin/dashboard?from=2025-01-01&to=2025-01-31
 */
export const getAdminDashboardHandler: RequestHandler = asyncHandler(
  async (req: Request, res: Response) => {
    const today = new Date();
    today.setHours(0, 0, 0, 0);

    const fromParam = req.query['from'] as string | undefined;
    const toParam = req.query['to'] as string | undefined;

    let from: Date;
    let to: Date;

    if (fromParam) {
      from = new Date(fromParam);
      if (isNaN(from.getTime())) throw AppError.badRequest('Invalid "from" date');
    } else {
      from = new Date(today);
      from.setDate(from.getDate() - 29);
    }

    if (toParam) {
      to = new Date(toParam);
      if (isNaN(to.getTime())) throw AppError.badRequest('Invalid "to" date');
    } else {
      to = today;
    }

    if (from > to) throw AppError.badRequest('"from" must be before or equal to "to"');

    const diffDays = Math.round((to.getTime() - from.getTime()) / 86_400_000);
    if (diffDays > 365) throw AppError.badRequest('Date range cannot exceed 365 days');

    const data = await getAdminDashboard(from, to);
    return ApiResponse.success(res, 'Admin dashboard data fetched successfully', data);
  },
);

/**
 * GET /api/v1/admin/users
 *
 * Returns all platform users with aggregated stats:
 *   totalSolved, currentStreak, acceptanceRate, rank, lastActive
 */
export const getAdminUsersHandler: RequestHandler = asyncHandler(
  async (_req: Request, res: Response) => {
    const data = await getAdminUsers();
    return ApiResponse.success(res, 'Users fetched successfully', data);
  },
);


/**
 * GET /api/v1/admin/users/:userId
 *
 * Returns full profile + stats for one user.
 */
export const getAdminUserDetailHandler: RequestHandler = asyncHandler(
  async (req: Request, res: Response) => {
    const userId = String(req.params['userId']);
    const data = await getAdminUserDetail(userId);
    return ApiResponse.success(res, 'User detail fetched successfully', data);
  },
);

/**
 * PATCH /api/v1/admin/users/:userId/role
 * Body: { role: "ADMIN" | "USER" }
 *
 * Promotes or demotes a user's role.
 */
export const updateUserRoleHandler: RequestHandler = asyncHandler(
  async (req: Request, res: Response) => {
    const userId = String(req.params['userId']);
    const { role } = req.body as { role: 'ADMIN' | 'USER' };
 
    if (role !== 'ADMIN' && role !== 'USER') {
      throw AppError.badRequest('role must be "ADMIN" or "USER"');
    }
    // Prevent self-demotion
    if (userId === req.user!.id && role === 'USER') {
      throw AppError.badRequest('You cannot remove your own admin role');
    }
 
    const data = await updateAdminUserRole(userId, role);
    return ApiResponse.success(
      res,
      role === 'ADMIN' ? 'User promoted to admin' : 'Admin role removed',
      data,
    );
  },
);

// src/controllers/auth.controller.ts
import type { Request, Response, RequestHandler } from 'express';
import jwt from 'jsonwebtoken';
import { ApiResponse } from '../utils/ApiResponse';
import { asyncHandler } from '../utils/asyncHandler';
import {
  registerUser,
  loginUser,
  getUserProfile,
  updateProfile,
  signToken,
} from '../services/auth.service';

const isDev = process.env.NODE_ENV !== 'production';

const COOKIE_OPTIONS = {
  httpOnly: true,
  // 'strict' blocks cookies on cross-origin requests (e.g. localhost:5173 → :8080)
  // Use 'lax' in development so the cookie is actually stored by the browser
  sameSite: (isDev ? 'lax' : 'strict') as 'lax' | 'strict',
  secure: !isDev, // only require HTTPS in production
  maxAge: 1000 * 60 * 60 * 24 * 7, // 7 days
};

export const registerHandler: RequestHandler = asyncHandler(async (req: Request, res: Response) => {
  const user = await registerUser(req.body);
  res.cookie('jwt', signToken(user.id), COOKIE_OPTIONS);
  return ApiResponse.created(res, 'User registered successfully', user);
});

export const loginHandler: RequestHandler = asyncHandler(async (req: Request, res: Response) => {
  const user = await loginUser(req.body);
  res.cookie('jwt', signToken(user.id), COOKIE_OPTIONS);
  return ApiResponse.success(res, 'Logged in successfully', user);
});

export const logoutHandler: RequestHandler = asyncHandler(async (_req: Request, res: Response) => {
  res.clearCookie('jwt', {
    httpOnly: true,
    sameSite: 'strict',
    secure: process.env.NODE_ENV !== 'development',
  });
  return ApiResponse.success(res, 'Logged out successfully');
});

export const checkHandler: RequestHandler = asyncHandler(async (req: Request, res: Response) => {
  return ApiResponse.success(res, 'Authenticated', req.user);
});

export const getUserDetailsHandler: RequestHandler = asyncHandler(
  async (req: Request, res: Response) => {
    const profile = await getUserProfile(req.user!.id);
    return ApiResponse.success(res, 'User details fetched successfully', profile);
  },
);

export const updateProfileHandler: RequestHandler = asyncHandler(
  async (req: Request, res: Response) => {
    const updated = await updateProfile(req.user!.id, req.body);
    return ApiResponse.success(res, 'Profile updated successfully', updated);
  },
);

export const googleCallbackHandler: RequestHandler = asyncHandler(
  async (req: Request, res: Response) => {
    const user = req.user as { id: string } | undefined;
    const clientUrl =
      process.env.NODE_ENV === 'production'
        ? process.env.CLIENT_PROD_URL
        : (process.env.CLIENT_DEV_URL ?? 'http://localhost:5173');

    if (!user) return res.redirect(`${clientUrl}/login?error=google_login_failed`);

    const token = jwt.sign({ id: user.id }, process.env.JWT_SECRET as string, { expiresIn: '7d' });
    res.cookie('jwt', token, {
      httpOnly: true,
      sameSite: 'lax',
      secure: process.env.NODE_ENV !== 'development',
      maxAge: 1000 * 60 * 60 * 24 * 7,
    });
    return res.redirect(`${clientUrl}/auth/google/callback?success=google_login`);
  },
);

// Used by passport session route — intentionally not asyncHandler
export const getCurrentUserHandler: RequestHandler = (req: Request, res: Response): void => {
  if (req.isAuthenticated()) {
    res.status(200).json({ success: true, data: req.user });
  } else {
    res.status(401).json({ success: false, message: 'Not authenticated' });
  }
};

// src/libs/formatMemory.ts

/**
 * Formats memory value from KB to a human-readable string.
 * @param memoryKB - Memory in KB (number or numeric string)
 */
export const formatMemory = (memoryKB: number | string | null | undefined): string | undefined => {
  if (memoryKB === undefined || memoryKB === null || memoryKB === '') return undefined;
  const kb = parseFloat(String(memoryKB));
  if (isNaN(kb)) return undefined;
  if (kb >= 1024) return `${(kb / 1024).toFixed(2)} MB`;
  return `${kb.toFixed(2)} KB`;
};

/**
 * Maps a raw Judge0 result + expected output into a normalised TestCaseDetail.
 * The _rawTime / _rawMemory fields are used for performance averaging and
 * must be stripped before sending to the client.
 */
export function mapJudge0Result(
  result: Judge0Result,
  index: number,
  expectedOutput: string,
): TestCaseDetail {
  const stdout = result.stdout?.trim() ?? '';
  return {
    testCase: index + 1,
    passed: stdout === expectedOutput.trim(),
    stdout,
    expected: expectedOutput,
    stderr: result.stderr ?? null,
    compile_output: result.compile_output ?? null,
    status: getStatusDescription(result),
    memory: result.memory ? (formatMemory(result.memory) ?? undefined) : undefined,
    time: result.time ? `${result.time} s` : undefined,
    _rawTime: result.time ? parseFloat(result.time) : null,
    _rawMemory: result.memory ? parseFloat(String(result.memory)) : null,
  };
}

/**
 * Averages runtime + memory across all test case results for a summary metric.
 */
export function buildPerformanceSummary(
  details: TestCaseDetail[],
): { totalTime: string; totalMemory: string } {
  const times = details
    .map((r) => r._rawTime)
    .filter((t): t is number => t !== null && !isNaN(t));
  const memories = details
    .map((r) => r._rawMemory)
    .filter((m): m is number => m !== null && !isNaN(m) && m > 0);

  const avgTime = times.length ? times.reduce((a, b) => a + b, 0) / times.length : null;
  const avgMemory =
    memories.length ? memories.reduce((a, b) => a + b, 0) / memories.length : null;

  return {
    totalTime: avgTime !== null ? `${avgTime.toFixed(3)} s` : '—',
    totalMemory: avgMemory !== null ? (formatMemory(avgMemory) ?? '—') : '—',
  };
}

// ─── Polling ──────────────────────────────────────────────────────────────────

const sleep = (ms: number): Promise<void> =>
  new Promise((resolve) => setTimeout(resolve, ms));

interface PollOptions {
  initialDelay?: number;
  pollInterval?: number;
  maxWaitTime?: number;
}

export const pollBatchResults = async (
  tokens: string[],
  {
    initialDelay = 200,
    pollInterval = 300,
    maxWaitTime = 30_000,
  }: PollOptions = {},
): Promise<Judge0Result[]> => {
  const tokenString = tokens.join(',');
  const startTime = Date.now();

  await sleep(initialDelay);

  while (true) {
    if (Date.now() - startTime > maxWaitTime) {
      throw new AppError('Judge0 polling timed out after 30s', 504);
    }

    try {
      const { data } = await judge0.get<{ submissions: Judge0Result[] }>(
        '/submissions/batch',
        {
          params: {
            tokens: tokenString,
            base64_encoded: false,
            fields: 'token,stdout,stderr,compile_output,status,memory,time',
          },
        },
      );

      const results = data?.submissions;
      if (!results?.length) {
        await sleep(pollInterval);
        continue;
      }

      const allDone = results.every((r) => r.status?.id !== 1 && r.status?.id !== 2);
      if (allDone) return results;
    } catch {
      // retry on transient network errors
    }

    await sleep(pollInterval);
  }
};

// ─── authMiddleware ───────────────────────────────────────────────────────────

export const authMiddleware: RequestHandler = asyncHandler(
  async (req: Request, _res: Response, next: NextFunction) => {
    const token = req.cookies?.jwt as string | undefined;
    if (!token) throw AppError.unauthorized('No token provided');

    let decoded: JwtPayload;
    try {
      decoded = jwt.verify(token, process.env.JWT_SECRET as string) as JwtPayload;
    } catch {
      throw AppError.unauthorized('Invalid or expired token');
    }

    const user = await db.user.findUnique({
      where: { id: decoded.id },
      select: { id: true, image: true, name: true, username: true, email: true, role: true },
    });

    if (!user) throw AppError.notFound('User not found');

    req.user = user as AuthUser;
    next();
  },
);

// ─── optionalAuth ─────────────────────────────────────────────────────────────

export const optionalAuth: RequestHandler = asyncHandler(
  async (req: Request, _res: Response, next: NextFunction) => {
    const token = req.cookies?.jwt as string | undefined;

    if (!token) {
      req.user = undefined;
      return next();
    }

    try {
      const decoded = jwt.verify(token, process.env.JWT_SECRET as string) as JwtPayload;
      const user = await db.user.findUnique({
        where: { id: decoded.id },
        select: { id: true, image: true, name: true, username: true, email: true, role: true },
      });
      req.user = (user as AuthUser) ?? undefined;
    } catch {
      req.user = undefined;
    }

    next();
  },
);

// src/middleware/errorHandler.ts
import type { Request, Response, NextFunction } from 'express';
import { Prisma } from '../generated/prisma/index.js';
import { AppError } from '../utils/AppError.js';
import { logger } from '../utils/logger.js';
import { ZodError } from 'zod';

// ─── Helpers ──────────────────────────────────────────────────────────────────

function errorBody(req: Request, statusCode: number, message: string, errors: unknown = null) {
  return {
    success: false,
    statusCode,
    message,
    timestamp: new Date().toISOString(),
    path: req.originalUrl,
    data: null,
    errors,
  };
}

// ─── Global error handler ────────────────────────────────────────────────────

export const globalErrorHandler = (
  err: unknown,
  req: Request,
  res: Response,
  _next: NextFunction,
): void => {
  // ── AppError (operational / known errors) ─────────────────────────────────
  if (err instanceof AppError) {
    const withErrors = err as AppError & { errors?: unknown };
    res
      .status(err.statusCode)
      .json(errorBody(req, err.statusCode, err.message, withErrors.errors ?? null));
    return;
  }

  // ── Zod validation errors ─────────────────────────────────────────────────
  if (err instanceof ZodError) {
    const errors = err.errors.map((e) => ({ field: e.path.join('.'), message: e.message }));
    res.status(400).json(errorBody(req, 400, 'Validation failed', errors));
    return;
  }

  // ── Prisma errors ─────────────────────────────────────────────────────────
  if (err instanceof Prisma.PrismaClientKnownRequestError) {
    if (err.code === 'P2002') {
      res.status(409).json(errorBody(req, 409, 'A record with this value already exists.'));
      return;
    }
    if (err.code === 'P2025') {
      res.status(404).json(errorBody(req, 404, 'Record not found.'));
      return;
    }
    logger.error('Prisma known request error:', err);
    res.status(500).json(errorBody(req, 500, 'Database error'));
    return;
  }

  if (err instanceof Prisma.PrismaClientValidationError) {
    res.status(400).json(errorBody(req, 400, 'Invalid data provided to database'));
    return;
  }

  // ── JWT errors ────────────────────────────────────────────────────────────
  if (err instanceof Error && err.name === 'JsonWebTokenError') {
    res.status(401).json(errorBody(req, 401, 'Invalid token'));
    return;
  }
  if (err instanceof Error && err.name === 'TokenExpiredError') {
    res.status(401).json(errorBody(req, 401, 'Token expired'));
    return;
  }

  // ── Unknown / programming errors ──────────────────────────────────────────
  logger.error('Unhandled error:', err);
  const message =
    process.env.NODE_ENV === 'production'
      ? 'Internal server error'
      : err instanceof Error
        ? err.message
        : 'Unknown error';
  res.status(500).json(errorBody(req, 500, message));
};

// ─── 404 catch-all ────────────────────────────────────────────────────────────

export const notFoundHandler = (req: Request, res: Response): void => {
  res.status(404).json(errorBody(req, 404, `Route ${req.originalUrl} not found`));
};

// src/middleware/validate.ts
import type { Request, Response, NextFunction } from 'express';
import type { ZodSchema } from 'zod';
import { AppError } from '../utils/AppError';

/**
 * Zod validation middleware.
 * Uses safeParse so we get structured field-level errors rather than a thrown
 * ZodError — the global error handler still catches thrown ZodErrors from
 * asyncHandler contexts, but this gives a consistent 400 shape for request
 * body validation specifically.
 */
export const validate =
  (schema: ZodSchema) =>
  (req: Request, _res: Response, next: NextFunction): void => {
    const result = schema.safeParse(req.body);

    if (!result.success) {
      const errors = result.error.errors.map((e) => ({
        field: e.path.join('.'),
        message: e.message,
      }));
      // Attach cleaned data and pass a structured error to the global handler
      return next(
        Object.assign(AppError.badRequest('Validation failed'), { errors }),
      );
    }

    // Replace req.body with the parsed (coerced + stripped) data
    req.body = result.data;
    next();
  };

function makeInitials(name: string | null, username: string | null): string {
  const source = name ?? username ?? '?';
  return source
    .split(' ')
    .map((w) => w[0] ?? '')
    .join('')
    .slice(0, 2)
    .toUpperCase();
}

/**
 * Returns BOTH current and max streak from an array of dates.
 * Used by getAdminUserDetail (needs both) and getAdminUsers (needs current only).
 */
function computeStreaks(dates: Date[]): { current: number; max: number } {
  if (!dates.length) return { current: 0, max: 0 };

  const unique = [...new Set(dates.map((d) => format(d, 'yyyy-MM-dd')))].sort((a, b) =>
    b.localeCompare(a),
  );

  const today = format(new Date(), 'yyyy-MM-dd');
  const yesterday = format(subDays(new Date(), 1), 'yyyy-MM-dd');

  // Current streak
  let current = 0;
  if (unique[0] === today || unique[0] === yesterday) {
    current = 1;
    for (let i = 1; i < unique.length; i++) {
      const expected = format(subDays(new Date(unique[i - 1]!), 1), 'yyyy-MM-dd');
      if (unique[i] === expected) current++;
      else break;
    }
  }

  // Max streak
  let max = unique.length > 0 ? 1 : 0;
  let temp = 1;
  for (let i = 1; i < unique.length; i++) {
    const expected = format(subDays(new Date(unique[i - 1]!), 1), 'yyyy-MM-dd');
    if (unique[i] === expected) {
      temp++;
      max = Math.max(max, temp);
    } else temp = 1;
  }

  return { current, max };
}

function buildDayMap(from: Date, to: Date): Map<string, DailyAnalyticsEntry> {
  const map = new Map<string, DailyAnalyticsEntry>();
  for (const day of eachDayOfInterval({ start: from, end: to })) {
    const key = format(day, 'yyyy-MM-dd');
    map.set(key, { date: format(day, 'MMM d'), signups: 0, active: 0, accepted: 0, rejected: 0 });
  }
  return map;
}

export async function getAdminDashboard(from: Date, to: Date): Promise<AdminDashboardData> {
  const rangeStart = startOfDay(from);
  const rangeEnd = endOfDay(to);
  const days = Math.round((rangeEnd.getTime() - rangeStart.getTime()) / 86_400_000) + 1;

  const priorEnd = subDays(rangeStart, 1);
  const priorStart = subDays(priorEnd, days - 1);

  const [
    signupRows,
    submissionRows,
    totalUsers,
    newUsersInRange,
    newUsersInPrior,
    problemCounts,
    sheetRows,
  ] = await Promise.all([
    db.user.findMany({
      where: { createdAt: { gte: rangeStart, lte: rangeEnd } },
      select: { createdAt: true },
    }),
    db.submission.findMany({
      where: { createdAt: { gte: rangeStart, lte: rangeEnd } },
      select: { createdAt: true, status: true, userId: true },
    }),
    db.user.count(),
    db.user.count({ where: { createdAt: { gte: rangeStart, lte: rangeEnd } } }),
    db.user.count({
      where: { createdAt: { gte: startOfDay(priorStart), lte: endOfDay(priorEnd) } },
    }),
    db.$transaction([
      db.problem.count(),
      db.problem.count({ where: { difficulty: 'EASY' } }),
      db.problem.count({ where: { difficulty: 'MEDIUM' } }),
      db.problem.count({ where: { difficulty: 'HARD' } }),
    ]),
    db.sheet.findMany({
      select: { id: true, name: true, description: true, _count: { select: { problems: true } } },
      orderBy: { createdAt: 'desc' },
    }),
  ]);

  const dayMap = buildDayMap(rangeStart, rangeEnd);

  for (const { createdAt } of signupRows) {
    const entry = dayMap.get(format(new Date(createdAt), 'yyyy-MM-dd'));
    if (entry) entry.signups += 1;
  }

  const activeSets = new Map<string, Set<string>>();
  for (const { createdAt, status, userId } of submissionRows) {
    const key = format(new Date(createdAt), 'yyyy-MM-dd');
    const entry = dayMap.get(key);
    if (!entry) continue;
    if (status === 'Accepted') entry.accepted += 1;
    else entry.rejected += 1;
    if (!activeSets.has(key)) activeSets.set(key, new Set());
    activeSets.get(key)!.add(userId);
  }
  for (const [key, set] of activeSets) {
    const entry = dayMap.get(key);
    if (entry) entry.active = set.size;
  }

  const analytics = Array.from(dayMap.values());
  const totalSubmissions = submissionRows.length;
  const totalActive = analytics.reduce((s, d) => s + d.active, 0);
  const avgDailyActive = days > 0 ? Math.round(totalActive / days) : 0;
  const userGrowthPct =
    newUsersInPrior > 0
      ? Math.round(((newUsersInRange - newUsersInPrior) / newUsersInPrior) * 100)
      : newUsersInRange > 0
        ? 100
        : 0;

  const [total, easy, medium, hard] = problemCounts as [number, number, number, number];

  return {
    analytics,
    summary: { days, totalUsers, newUsersInRange, userGrowthPct, avgDailyActive, totalSubmissions },
    problemStats: { total, easy, medium, hard },
    sheets: sheetRows.map((s) => ({
      id: s.id,
      name: s.name,
      description: s.description,
      totalProblems: s._count.problems,
    })),
  };
}

export async function getAdminUsers(): Promise<AdminUsersData> {
  const [users, allSolved, submissionDates, allSubmissions] = await Promise.all([
    db.user.findMany({
      select: { id: true, name: true, username: true, email: true, role: true, image: true },
      orderBy: { createdAt: 'desc' },
    }),
    db.problemSolved.findMany({
      select: { userId: true, problemId: true, createdAt: true },
    }),
    db.submission.findMany({
      select: { userId: true, createdAt: true },
      orderBy: { createdAt: 'desc' },
    }),
    db.submission.findMany({
      select: { userId: true, status: true },
    }),
  ]);

  // Solved map
  const solvedMap = new Map<string, { problems: Set<string>; dates: Date[] }>();
  for (const { userId, problemId, createdAt } of allSolved) {
    if (!solvedMap.has(userId)) solvedMap.set(userId, { problems: new Set(), dates: [] });
    const e = solvedMap.get(userId)!;
    e.problems.add(problemId);
    e.dates.push(createdAt);
  }

  // Last active map
  const lastActiveMap = new Map<string, Date>();
  for (const { userId, createdAt } of submissionDates) {
    if (!lastActiveMap.has(userId)) lastActiveMap.set(userId, createdAt);
  }

  // Submission counts
  const totalSubMap = new Map<string, number>();
  const acceptedSubMap = new Map<string, number>();
  for (const { userId, status } of allSubmissions) {
    totalSubMap.set(userId, (totalSubMap.get(userId) ?? 0) + 1);
    if (status === 'Accepted') acceptedSubMap.set(userId, (acceptedSubMap.get(userId) ?? 0) + 1);
  }

  // Global rank
  const ranked = [...solvedMap.entries()]
    .map(([userId, { problems }]) => ({ userId, count: problems.size }))
    .sort((a, b) => b.count - a.count);
  const rankMap = new Map<string, number>();
  ranked.forEach(({ userId }, i) => rankMap.set(userId, i + 1));

  let totalSolvedSum = 0;

  const result: AdminUserEntry[] = users.map((u) => {
    const solved = solvedMap.get(u.id);
    const totalSolved = solved?.problems.size ?? 0;
    totalSolvedSum += totalSolved;

    // ✅ Destructure .current — never assign the object itself
    const { current: currentStreak } = solved ? computeStreaks(solved.dates) : { current: 0 };

    const total = totalSubMap.get(u.id) ?? 0;
    const accepted = acceptedSubMap.get(u.id) ?? 0;
    const acceptanceRate = total > 0 ? Math.round((accepted / total) * 100) : 0;

    const lastActiveDate = lastActiveMap.get(u.id);
    const lastActive = lastActiveDate ? format(lastActiveDate, 'MMM d, yyyy') : null;

    return {
      id: u.id,
      displayName: u.name ?? u.username ?? u.email,
      username: u.username ?? u.email.split('@')[0]!,
      email: u.email,
      initials: makeInitials(u.name, u.username),
      role: u.role as 'ADMIN' | 'USER',
      image: u.image,
      totalSolved,
      currentStreak, // ← plain number, not an object
      acceptanceRate,
      rank: rankMap.get(u.id) ?? users.length,
      lastActive,
    };
  });

  return { users: result, stats: { total: users.length, totalSolved: totalSolvedSum } };
}

export async function getAdminUserDetail(userId: string): Promise<AdminUserDetail> {
  const [user, solvedRecords, allSubmissions, latestSubmission, diffTotals, allTags] =
    await Promise.all([
      db.user.findUnique({
        where: { id: userId },
        select: {
          id: true,
          name: true,
          username: true,
          email: true,
          role: true,
          image: true,
          createdAt: true,
        },
      }),
      db.problemSolved.findMany({
        where: { userId },
        select: { createdAt: true, problem: { select: { difficulty: true, tags: true } } },
      }),
      db.submission.findMany({
        where: { userId },
        orderBy: { createdAt: 'desc' },
        select: {
          id: true,
          status: true,
          language: true,
          createdAt: true,
          problem: { select: { id: true, title: true } },
          testCases: { select: { time: true } },
        },
      }),
      db.submission.findFirst({
        where: { userId },
        orderBy: { createdAt: 'desc' },
        select: { createdAt: true },
      }),
      db.$transaction([
        db.problem.count({ where: { difficulty: 'EASY' } }),
        db.problem.count({ where: { difficulty: 'MEDIUM' } }),
        db.problem.count({ where: { difficulty: 'HARD' } }),
      ]),
      db.problem.findMany({ select: { tags: true } }),
    ]);

  if (!user) throw AppError.notFound('User not found');

  // Difficulty breakdown
  let easySolved = 0,
    mediumSolved = 0,
    hardSolved = 0;
  const solvedByTag = new Map<string, number>();
  const solvedDates: Date[] = [];

  for (const record of solvedRecords) {
    const diff = record.problem.difficulty;
    if (diff === 'EASY') easySolved++;
    else if (diff === 'MEDIUM') mediumSolved++;
    else if (diff === 'HARD') hardSolved++;
    for (const tag of record.problem.tags) {
      solvedByTag.set(tag, (solvedByTag.get(tag) ?? 0) + 1);
    }
    solvedDates.push(record.createdAt);
  }

  const totalSolved = solvedRecords.length;
  const totalSubmissions = allSubmissions.length;
  const acceptedCount = allSubmissions.filter((s) => s.status === 'Accepted').length;
  const acceptanceRate =
    totalSubmissions > 0 ? Math.round((acceptedCount / totalSubmissions) * 100) : 0;

  // ✅ Destructure both values cleanly
  const { current: currentStreak, max: maxStreak } = computeStreaks(solvedDates);

  // Global rank
  const usersAhead = await db.problemSolved.groupBy({
    by: ['userId'],
    _count: { problemId: true },
    having: { problemId: { _count: { gt: totalSolved } } },
  });
  const rank = usersAhead.length + 1;

  // Platform tag totals
  const platformTagCounts = new Map<string, number>();
  for (const { tags } of allTags) {
    for (const tag of tags) platformTagCounts.set(tag, (platformTagCounts.get(tag) ?? 0) + 1);
  }

  const topTopics = [...solvedByTag.entries()]
    .map(([topic, solved]) => ({ topic, solved, total: platformTagCounts.get(topic) ?? solved }))
    .sort((a, b) => b.solved - a.solved)
    .slice(0, 10);

  const recentSubmissions = allSubmissions.slice(0, 10).map((s) => {
    const totalTime = s.testCases.reduce((acc, tc) => acc + (parseFloat(tc.time ?? '0') || 0), 0);
    return {
      problem: s.problem.title,
      problemId: s.problem.id,
      status: s.status,
      language: s.language,
      runtime: `${totalTime.toFixed(2)}ms`,
      date: format(s.createdAt, 'MMM d, yyyy'),
    };
  });

  return {
    id: user.id,
    displayName: user.name ?? user.username ?? user.email,
    username: user.username ?? user.email.split('@')[0]!,
    email: user.email,
    initials: makeInitials(user.name, user.username),
    image: user.image,
    role: user.role as 'ADMIN' | 'USER',
    joinDate: format(user.createdAt, 'MMM yyyy'),
    lastActive: latestSubmission ? format(latestSubmission.createdAt, 'MMM d, yyyy') : null,
    totalSolved,
    easySolved,
    mediumSolved,
    hardSolved,
    totalSubmissions,
    acceptanceRate,
    currentStreak,
    maxStreak,
    rank,
    difficultyTotals: {
      easy: (diffTotals as [number, number, number])[0],
      medium: (diffTotals as [number, number, number])[1],
      hard: (diffTotals as [number, number, number])[2],
    },
    topTopics,
    recentSubmissions,
  };
}

export async function updateAdminUserRole(
  userId: string,

async function generateUniqueUsername(baseUsername: string): Promise<string> {
  const MAX_ATTEMPTS = 50;
  const similar = await db.user.findMany({
    where: { username: { startsWith: baseUsername } },
    select: { username: true },
  });
  const existing = new Set(
    similar
      .map((u: { username: string | null }) => u.username)
      .filter((u: string | null): u is string => u !== null),
  );
  let username = baseUsername;
  let suffix = 0;
  while (existing.has(username)) {
    suffix++;
    if (suffix > MAX_ATTEMPTS) {
      // Fall back to a random suffix to guarantee uniqueness
      username = `${baseUsername}_${Date.now()}`;
      break;
    }
    username = `${baseUsername}_${suffix}`;
  }
  return username;
}

export async function runCode(input: RunCodeInput): Promise<RunResult> {
  const { source_code, language_id, problemId } = input;

  const publicTestCases = await db.testCase.findMany({
    where: { problemId, isPublic: true },
    orderBy: { id: 'asc' },
    select: { input: true, expected: true },
  });

  if (!publicTestCases.length) throw AppError.notFound('No public test cases found');

  const tokens = (
    await submitBatch(
      publicTestCases.map(({ input }) => ({ source_code, language_id, stdin: input })),
    )
  ).map((r) => r.token);

  const results = await pollBatchResults(tokens);
  const details = results.map((r, i) =>
    mapJudge0Result(r, i, publicTestCases[i]?.expected ?? ''),
  );

  return {
    allPassed: details.every((d) => d.passed),
    performance: buildPerformanceSummary(details),
    results: details.map(({ _rawTime, _rawMemory, ...rest }) => rest),
  };
}

// ─── Submit (all test cases) ──────────────────────────────────────────────────

export async function submitCode(input: SubmitCodeInput): Promise<SubmitResult> {
  const { source_code, language_id, problemId, userId } = input;

  const testCases = await db.testCase.findMany({
    where: { problemId },
    orderBy: { isPublic: 'desc' },
    select: { id: true, input: true, expected: true, isPublic: true },
  });

  if (!testCases.length) throw AppError.notFound('No test cases found for this problem');

  const tokens = (
    await submitBatch(
      testCases.map(({ input }) => ({ source_code, language_id, stdin: input })),
    )
  ).map((r) => r.token);

  const results = await pollBatchResults(tokens);

  const details: TestCaseDetail[] = results.map((result, i) => {
    const tc = testCases[i];
    const base = mapJudge0Result(result, i, tc?.expected ?? '');
    return {
      ...base,
      testCaseId: tc?.id,
      isPublic: tc?.isPublic,
      expected: tc?.isPublic ? base.expected : '', // hide expected for private cases
    };
  });

  const allPassed = details.every((d) => d.passed);
  const language = getLanguageName(language_id);

  // Persist submission — sourceCode is Json in schema
  const submission = await db.submission.create({
    data: {
      userId,
      problemId,
      sourceCode: { code: source_code, language },
      language,
      stdin: testCases.map((tc) => tc.input).join('\n'),
      stdout: JSON.stringify(details.map((r) => r.stdout)),
      stderr: details.some((r) => r.stderr)
        ? JSON.stringify(details.map((r) => r.stderr))
        : null,
      compileOutput: details.some((r) => r.compile_output)
        ? JSON.stringify(details.map((r) => r.compile_output))
        : null,
      status: allPassed ? 'Accepted' : 'Wrong Answer',
      memory: JSON.stringify(details.map((r) => r.memory)),
      time: JSON.stringify(details.map((r) => r.time)),
    },
  });

  await Promise.all([
    db.testCaseResult.createMany({
      data: details.map((d) => ({
        submissionId: submission.id,
        testCase: d.testCase,
        passed: d.passed,
        stdout: d.stdout ?? '',
        expected: d.expected ?? '', // non-nullable in schema
        stderr: d.stderr,
        compileOutput: d.compile_output,
        status: d.status,
        memory: d.memory,
        time: d.time,
      })),
    }),
    allPassed
      ? db.problemSolved.upsert({
          where: { userId_problemId: { userId, problemId } },
          update: {},
          create: { userId, problemId },
        })
      : Promise.resolve(),
  ]);

  const passedCount = details.filter((r) => r.passed).length;

  return {
    allPassed,
    submission: {
      language,
      status: allPassed ? 'Accepted' : 'Wrong Answer',
      performance: buildPerformanceSummary(details),
      testCasesPassed: `${passedCount}/${details.length}`,
    },
  };
}

export async function createComment(input: CreateCommentInput) {
  const { userId, discussionId, content, parentId } = input;

  const discussion = await db.discussion.findUnique({ where: { id: discussionId } });
  if (!discussion) throw AppError.notFound('Discussion not found');

  if (parentId) {
    const parent = await db.comment.findUnique({ where: { id: parentId } });
    if (!parent) throw AppError.notFound('Parent comment not found');
    if (parent.discussionId !== discussionId) {
      throw AppError.badRequest('Parent comment must belong to the same discussion');
    }

    // Enforce max nesting depth of 3
    let depth = 1;
    let currentParentId = parent.parentId;
    while (currentParentId) {
      if (depth >= 3) throw AppError.badRequest('Maximum comment nesting depth reached');
      const ancestor = await db.comment.findUnique({
        where: { id: currentParentId },
        select: { parentId: true },
      });
      if (!ancestor) break;
      currentParentId = ancestor.parentId;
      depth++;
    }
  }

  return db.comment.create({
    data: { content: content.trim(), discussionId, parentId: parentId ?? null, authorId: userId },
    include: COMMENT_INCLUDE,
  });
}

  const whereClause = isDiscussion
    ? { userId_discussionId: { userId, discussionId: entityId } }
    : { userId_commentId: { userId, commentId: entityId } };

  const createData = isDiscussion
    ? { userId, discussionId: entityId, value }
    : { userId, commentId: entityId, value };

  if (value === 0) {
    await db.vote.deleteMany({
      where: isDiscussion ? { userId, discussionId: entityId } : { userId, commentId: entityId },
    });
  } else {
    await db.vote.upsert({ where: whereClause, create: createData, update: { value } });
  }

  const votes = await db.vote.findMany({
    where: isDiscussion ? { discussionId: entityId } : { commentId: entityId },
    select: { userId: true, value: true },
  });

  return {
    upvotes: votes.filter((v) => v.value === 1).length,
    downvotes: votes.filter((v) => v.value === -1).length,

function computeStreak(dates: Date[]): number {
  if (!dates.length) return 0;

  const uniqueDays = [
    ...new Set(dates.map((d) => format(new Date(d), 'yyyy-MM-dd'))),
  ].sort((a, b) => b.localeCompare(a));

  const today = format(new Date(), 'yyyy-MM-dd');
  const yesterday = format(subDays(new Date(), 1), 'yyyy-MM-dd');

  const first = uniqueDays[0];
  if (first !== today && first !== yesterday) return 0;

  let streak = 1;
  for (let i = 1; i < uniqueDays.length; i++) {
    const prev = uniqueDays[i - 1];
    const curr = uniqueDays[i];
    if (!prev || !curr) break;
    const expected = format(subDays(new Date(prev), 1), 'yyyy-MM-dd');
    if (curr === expected) streak++;
    else break;
  }
  return streak;
}

// ─── Service Method ───────────────────────────────────────────────────────────

export async function getLeaderboard(options: LeaderboardOptions = {}): Promise<LeaderboardResponse> {
  const limit = Math.min(options.limit ?? 50, 100); // cap at 100
  const offset = options.offset ?? 0;

  const allSolved = await db.problemSolved.findMany({
    select: { userId: true, problemId: true, createdAt: true },
  });

  // Aggregate per user
  const userMap = new Map<string, { problems: Set<string>; dates: Date[] }>();
  for (const { userId, problemId, createdAt } of allSolved) {
    if (!userMap.has(userId)) userMap.set(userId, { problems: new Set(), dates: [] });
    const entry = userMap.get(userId)!;
    entry.problems.add(problemId);
    entry.dates.push(createdAt);
  }

  const ranked = Array.from(userMap.entries())
    .map(([userId, { problems, dates }]) => {
      const solved = problems.size;
      const streak = computeStreak(dates);
      return { userId, solved, streak, score: solved * 50 + streak * 10 };
    })
    .sort((a, b) => b.score - a.score || b.solved - a.solved);

  const total = ranked.length;
  const page = ranked.slice(offset, offset + limit);

  // Batch-fetch only the users we need for this page
  const userIds = page.map((e) => e.userId);
  const users = await db.user.findMany({
    where: { id: { in: userIds } },
    select: { id: true, username: true, name: true, image: true },
  });

  const userInfoMap = new Map(
    users.map((u) => [u.id, {
      username: u.username ?? u.name ?? 'Unknown',
      name: u.name ?? u.username ?? 'Unknown',
      image: u.image ?? null,
    }]),
  );

  const data: LeaderboardEntry[] = page.map((entry, i) => ({
    rank: offset + i + 1,
    username: userInfoMap.get(entry.userId)?.username ?? 'Unknown',
    name: userInfoMap.get(entry.userId)?.name ?? 'Unknown',
    image: userInfoMap.get(entry.userId)?.image ?? null,
    solved: entry.solved,
    streak: entry.streak,
    score: entry.score,
  }));

  return { data, total, limit, offset };
}

function decodeCodeString(code: string): string {
  return code
    .replace(/\\n/g, '\n')
    .replace(/\\t/g, '\t')
    .replace(/\\"/g, '"')
    .replace(/\\\\/g, '\\');
}

function sanitizeCompanyTags(tags: unknown): string[] {
  if (!Array.isArray(tags)) return [];
  return tags
    .filter((t): t is string => typeof t === 'string' && t.trim().length > 0)
    .map((t) => t.trim())
    .slice(0, 20);
}

export async function createProblem(input: CreateProblemInput) {
  const { problemNumber, problem, testCases, userId } = input;
  const {
    title,
    description,
    difficulty,
    tags = [],
    examples = {},
    constraints = '',
    codeSnippets = {},
    referenceSolutions = {},
    hints = '',
    editorial = '',
    companyTags = [],
  } = problem;

  const existingProblem = await db.problem.findFirst({
    where: { OR: [{ title }, ...(problemNumber ? [{ problemNumber }] : [])] },
  });
  if (existingProblem) {
    throw AppError.conflict('Problem with this title or number already exists');
  }

  // Validate every reference solution against all test cases via Judge0
  for (const [language, solutionCodeRaw] of Object.entries(referenceSolutions)) {
    const solutionCode = decodeCodeString(solutionCodeRaw as string);
    const languageId = getJudge0LanguageId(language);
    if (!languageId) throw AppError.badRequest(`Language ${language} is not supported`);

    const judgeSubmissions = testCases.map(({ input, expected }) => ({
      source_code: solutionCode,
      language_id: languageId,
      stdin: input,
      expected_output: expected,
    }));

    const submitResponse = await submitBatch(judgeSubmissions);
    const tokens = submitResponse.map((r) => r.token);
    const results = await pollBatchResults(tokens);

    for (let i = 0; i < results.length; i++) {
      if (results[i]?.status.id !== 3) {
        throw AppError.badRequest(
          `Reference solution failed on test case #${i + 1} for language ${language}`,
        );
      }
    }
  }

  return db.$transaction(async (tx: Parameters<Parameters<typeof db.$transaction>[0]>[0]) => {
    const created = await tx.problem.create({
      data: {
        problemNumber,
        title,
        description,
        difficulty: difficulty as 'EASY' | 'MEDIUM' | 'HARD',
        tags,
        examples: examples as Prisma.InputJsonValue,
        constraints,
        codeSnippets,
        referenceSolutions,
        userId,
        hints,
        editorial,
        companyTags: sanitizeCompanyTags(companyTags),
      },
    });

    if (testCases.length > 0) {
      await tx.testCase.createMany({
        data: testCases.map((tc) => ({
          input: tc.input,
          expected: tc.expected,
          isPublic: Boolean(tc.isPublic),
          problemId: created.id,
        })),
      });
    }

    return created;
  });

function convertToKB(memStr: string | number | null | undefined): number {
  if (typeof memStr === 'number') return memStr;
  if (!memStr) return 0;
  const match = String(memStr).match(/([\d.]+)\s*(KB|MB|GB)?/i);
  if (!match) return 0;
  const value = parseFloat(match[1] ?? '0');
  const unit = (match[2] ?? 'KB').toUpperCase();
  if (unit === 'MB') return value * 1024;
  if (unit === 'GB') return value * 1024 * 1024;
  return value;
}

// ─── Helpers ──────────────────────────────────────────────────────────────────

function formatTimeAgo(date: Date): string {
  const diffHrs = Math.floor((Date.now() - new Date(date).getTime()) / (1000 * 60 * 60));
  if (diffHrs < 1) return 'Just now';
  if (diffHrs < 24) return `${diffHrs}h ago`;
  const diffDays = Math.floor(diffHrs / 24);
  if (diffDays < 7) return `${diffDays}d ago`;
  return format(new Date(date), 'MMM d');
}

function parseAvg(raw: string | null, unit: string): string {
  if (!raw) return '-';
  try {
    const parsed = JSON.parse(raw);
    const arr = Array.isArray(parsed) ? parsed : [parsed];
    const nums = arr.map((s: unknown) => parseFloat(String(s))).filter((n) => !isNaN(n));
    if (!nums.length) return '-';
    const avg = nums.reduce((a, b) => a + b, 0) / nums.length;
    return `${avg.toFixed(3)}${unit}`;
  } catch {
    const num = parseFloat(String(raw));
    return isNaN(num) ? '-' : `${num.toFixed(3)}${unit}`;
  }
}

function computeStreaks(uniqueDates: string[]): { current: number; max: number } {
  if (!uniqueDates.length) return { current: 0, max: 0 };

  const today = format(new Date(), 'yyyy-MM-dd');
  const yesterday = format(subDays(new Date(), 1), 'yyyy-MM-dd');

  let current = 0;
  if (uniqueDates[0]! === today || uniqueDates[0]! === yesterday) {
    current = 1;
    for (let i = 1; i < uniqueDates.length; i++) {
      const expected = format(subDays(new Date(uniqueDates[i - 1]!), 1), 'yyyy-MM-dd');
      if (uniqueDates[i]! === expected) current++;
      else break;
    }
  }

  let max = uniqueDates.length > 0 ? 1 : 0;
  let temp = 1;
  for (let i = 1; i < uniqueDates.length; i++) {
    const expected = format(subDays(new Date(uniqueDates[i - 1]!), 1), 'yyyy-MM-dd');
    if (uniqueDates[i]! === expected) {
      temp++;
      max = Math.max(max, temp);
    } else {
      temp = 1;
    }
  }

  return { current, max };
}

export async function getHeatmap(userId: string): Promise<HeatmapDay[]> {
  const startDate = subDays(new Date(), 364);

  const submissions = await db.submission.findMany({
    where: { userId, createdAt: { gte: startDate } },
    select: { createdAt: true },
  });

  const countByDate: Record<string, number> = {};
  submissions.forEach((s) => {
    const key = format(new Date(s.createdAt), 'yyyy-MM-dd');
    countByDate[key] = (countByDate[key] ?? 0) + 1;
  });

  const today = new Date();
  return Array.from({ length: 365 }, (_, i) => {
    const date = subDays(today, i);
    const formatted = format(date, 'yyyy-MM-dd');
    return { date: formatted, count: countByDate[formatted] ?? 0 };
  }).reverse();
}

export async function getProgressData(userId: string): Promise<ProgressData> {
  const end = endOfDay(new Date());
  const weekStart = startOfDay(subDays(new Date(), 6));

  // Build month boundaries for last 6 months
  const monthRanges = Array.from({ length: 6 }, (_, i) => {
    const d = new Date();
    d.setMonth(d.getMonth() - (5 - i));
    return {
      label: format(d, 'MMM'),
      start: new Date(d.getFullYear(), d.getMonth(), 1),
      end: new Date(d.getFullYear(), d.getMonth() + 1, 0, 23, 59, 59),
    };
  });

  // Fire all DB queries in a single Promise.all — no sequential awaits
  const [recentSubmissions, allSubmissions, latestSubmissions, ...monthCounts] = await Promise.all([
    db.submission.findMany({
      where: { userId, createdAt: { gte: weekStart, lte: end } },
      select: { createdAt: true },
    }),
    db.submission.findMany({
      where: { userId },
      select: { language: true },
    }),
    db.submission.findMany({
      where: { userId },
      orderBy: { createdAt: 'desc' },
      take: 10,
      select: {
        status: true,
        language: true,
        memory: true,
        time: true,
        createdAt: true,
        problem: { select: { title: true, difficulty: true, id: true } },
      },
    }),
    // 6 × 2 count queries run in parallel
    ...monthRanges.flatMap(({ start, end: mEnd }) => [
      db.submission.count({ where: { userId, createdAt: { gte: start, lte: mEnd } } }),
      db.problemSolved.count({ where: { userId, createdAt: { gte: start, lte: mEnd } } }),
    ]),
  ]);

  // Weekly activity
  const submissionsByDate: Record<string, number> = {};
  recentSubmissions.forEach((item) => {
    const key = format(item.createdAt, 'yyyy-MM-dd');
    submissionsByDate[key] = (submissionsByDate[key] ?? 0) + 1;
  });

  const weeklyData: WeeklyDay[] = Array.from({ length: 7 }, (_, i) => {
    const date = subDays(new Date(), 6 - i);
    const key = format(date, 'yyyy-MM-dd');
    return { date: key, day: format(date, 'EEE'), problems: submissionsByDate[key] ?? 0 };
  });

  // Monthly trend — unpack pairs from flat monthCounts array
  const monthlyTrend: MonthlyEntry[] = monthRanges.map(({ label }, i) => ({
    month: label,
    submissions: (monthCounts[i * 2] as number) ?? 0,
    solved: (monthCounts[i * 2 + 1] as number) ?? 0,
  }));

  // Language breakdown
  const languageCounts: Record<string, number> = {};
  allSubmissions.forEach((s) => {
    languageCounts[s.language] = (languageCounts[s.language] ?? 0) + 1;
  });

  const totalLangCount = Object.values(languageCounts).reduce((a, b) => a + b, 0);
  const languageStats: LanguageStat[] = Object.entries(languageCounts)
    .map(([language, count]) => ({
      language,
      count,
      percentage: totalLangCount > 0 ? Math.round((count / totalLangCount) * 100) : 0,
    }))
    .sort((a, b) => b.count - a.count);

  const formattedRecent: RecentSubmission[] = latestSubmissions.map((s) => ({
    problem: s.problem.title,
    problemId: s.problem.id,
    difficulty: s.problem.difficulty.charAt(0) + s.problem.difficulty.slice(1).toLowerCase(),
    status: s.status,
    language: s.language,
    time: parseAvg(s.time, 's'),
    memory: parseAvg(s.memory, ' MB'),
    date: formatTimeAgo(s.createdAt),
  }));

  return {
    weeklyData,
    monthlyTrend,
    languageStats,
    recentSubmissions: formattedRecent,
  };
}

export async function getSolvedStats(userId: string): Promise<SolvedStats> {
  const solvedProblems = await db.problemSolved.findMany({
    where: { userId },
    select: { problemId: true },
    distinct: ['problemId'],
  });

  const solvedProblemIds = solvedProblems.map((item) => item.problemId);

  const [solvedDetails, allProblems] = await Promise.all([
    solvedProblemIds.length > 0
      ? db.problem.findMany({
          where: { id: { in: solvedProblemIds } },
          select: { difficulty: true, tags: true },
        })
      : Promise.resolve([]),
    db.problem.findMany({ select: { difficulty: true, tags: true } }),
  ]);

  const difficultyStats: Record<string, number> = { EASY: 0, MEDIUM: 0, HARD: 0 };
  const tagStats: Record<string, number> = {};
  for (const p of solvedDetails) {
    difficultyStats[p.difficulty] = (difficultyStats[p.difficulty] ?? 0) + 1;
    for (const tag of p.tags) tagStats[tag] = (tagStats[tag] ?? 0) + 1;
  }

  const totalDifficultyCounts: Record<string, number> = { EASY: 0, MEDIUM: 0, HARD: 0 };
  const totalTagCounts: Record<string, number> = {};
  for (const p of allProblems) {
    totalDifficultyCounts[p.difficulty] = (totalDifficultyCounts[p.difficulty] ?? 0) + 1;
    for (const tag of p.tags) totalTagCounts[tag] = (totalTagCounts[tag] ?? 0) + 1;
  }

  // Streak
  const allSolvedDates = await db.problemSolved.findMany({
    where: { userId },
    select: { createdAt: true },
    orderBy: { createdAt: 'desc' },
  });

  const uniqueDates: string[] = Array.from<string>(
    new Set<string>(allSolvedDates.map((s) => format(new Date(s.createdAt), 'yyyy-MM-dd'))),
  ).sort((a, b) => b.localeCompare(a));

  const { current: currentStreak, max: maxStreak } = computeStreaks(uniqueDates);

  const [totalSubmissions, acceptedSubmissions] = await Promise.all([
    db.submission.count({ where: { userId } }),
    db.submission.count({ where: { userId, status: 'Accepted' } }),
  ]);

  const acceptanceRate =
    totalSubmissions > 0 ? Math.round((acceptedSubmissions / totalSubmissions) * 100) : 0;

  return {
    difficultyStats,
    tagStats,
    totalDifficultyCounts,
    totalTagCounts,
    currentStreak,
    maxStreak,
    totalSubmissions,
    acceptanceRate,
  };
}

// src/utils/ApiResponse.ts
import type { Request, Response } from 'express';

interface SuccessPayload<T> {
  success: true;
  statusCode: number;
  message: string;
  timestamp: string;
  data: T | null;
}

interface ErrorPayload {
  success: false;
  statusCode: number;
  message: string;
  timestamp: string;
  path: string;
  data: null;
  errors: unknown | null;
}

export class ApiResponse {
  // ─── Success ────────────────────────────────────────────────────────────────

  static success<T>(res: Response, message: string, data?: T, statusCode = 200): Response {
    const payload: SuccessPayload<T> = {
      success: true,
      statusCode,
      message,
      timestamp: new Date().toISOString(),
      data: data ?? null,
    };
    return res.status(statusCode).json(payload);
  }

  static created<T>(res: Response, message: string, data?: T): Response {
    return ApiResponse.success(res, message, data, 201);
  }

  // ─── Error ──────────────────────────────────────────────────────────────────

  static error(
    res: Response,
    req: Request,
    message: string,
    statusCode = 500,
    errors?: unknown,
  ): Response {
    const payload: ErrorPayload = {
      success: false,
      statusCode,
      message,
      timestamp: new Date().toISOString(),
      path: req.originalUrl,
      data: null,
      errors: errors ?? null,
    };
    return res.status(statusCode).json(payload);
  }

  static badRequest(res: Response, req: Request, message: string, errors?: unknown): Response {
    return ApiResponse.error(res, req, message, 400, errors);
  }

  static unauthorized(res: Response, req: Request, message = 'Unauthorized'): Response {
    return ApiResponse.error(res, req, message, 401);
  }

  static forbidden(res: Response, req: Request, message = 'Forbidden'): Response {
    return ApiResponse.error(res, req, message, 403);
  }

  static notFound(res: Response, req: Request, message = 'Not found'): Response {
    return ApiResponse.error(res, req, message, 404);
  }

  static conflict(res: Response, req: Request, message: string): Response {
    return ApiResponse.error(res, req, message, 409);
  }

  static serverError(res: Response, req: Request, message = 'Internal server error'): Response {
    return ApiResponse.error(res, req, message, 500);
  }
}

  res: Response,
  next: NextFunction,
) => Promise<unknown>;

/**

 * to Express's global error handler via next(err).

  (req, res, next) => {
